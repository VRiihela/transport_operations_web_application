## Task Specification

**Description:** Implement secure authentication and role-based access control (RBAC) for a small transport company work management system. Users sign in with email and password. There are three roles: Admin, Dispatcher, and Driver. Drivers may only access their own assigned jobs. Dispatchers and Admins can access operational dashboards and all jobs. Use JWT access tokens (short-lived) and refresh tokens (long-lived, stored securely). Passwords must be hashed with bcrypt. All auth-related routes must be protected against common attacks.

**Acceptance Criteria:**
- Users can sign in with email and password and receive a JWT access token and refresh token
- Access tokens expire after 15 minutes; refresh tokens expire after 7 days
- Refresh token endpoint issues a new access token and rotates the refresh token
- Sign-out invalidates the refresh token
- Passwords are hashed with bcrypt (minimum cost factor 12)
- Role-based middleware enforces access: Admin and Dispatcher can access all jobs, Driver can only access own assigned jobs
- All inputs (email, password) are validated server-side with Zod
- Auth errors return generic messages (no user enumeration)
- Tokens are not stored in localStorage — refresh token is httpOnly cookie
- Protected routes return 401 if no valid token, 403 if insufficient role

import jwt from 'jsonwebtoken';
import bcrypt from 'bcrypt';
import { JwtPayload, RefreshToken, UserRole } from '../models/user';
import { auditLogger } from '../utils/auditLogger';

export class TokenService {
  private readonly JWT_SECRET: string;
  private readonly REFRESH_SECRET: string;
  private readonly ACCESS_TOKEN_EXPIRY = '15m';
  private readonly REFRESH_TOKEN_EXPIRY = '7d';

  constructor() {
    this.JWT_SECRET = process.env.JWT_SECRET!;
    this.REFRESH_SECRET = process.env.REFRESH_SECRET!;
    
    if (!this.JWT_SECRET || !this.REFRESH_SECRET) {
      throw new Error('JWT_SECRET and REFRESH_SECRET environment variables are required');
    }
  }

  generateAccessToken(userId: number, email: string, role: UserRole): string {
    const payload: Omit<JwtPayload, 'iat' | 'exp'> = {
      userId,
      email,
      role
    };

    return jwt.sign(payload, this.JWT_SECRET, {
      expiresIn: this.ACCESS_TOKEN_EXPIRY,
      issuer: 'work-management-system',
      audience: 'work-management-users'
    });
  }

  generateRefreshToken(): string {
    // Generate a random 64-character hex string
    const randomBytes = require('crypto').randomBytes(32);
    return randomBytes.toString('hex');
  }

  async hashRefreshToken(token: string): Promise<string> {
    return bcrypt.hash(token, 12);
  }

  async validateRefreshToken(token: string, hashedToken: string): Promise<boolean> {
    try {
      return await bcrypt.compare(token, hashedToken);
    } catch (error) {
      auditLogger.logSecurityEvent('REFRESH_TOKEN_VALIDATION_ERROR', null, false, { error: 'Token validation failed' });
      return false;
    }
  }

  verifyAccessToken(token: string): JwtPayload | null {
    try {
      const decoded = jwt.verify(token, this.JWT_SECRET, {
        issuer: 'work-management-system',
        audience: 'work-management-users'
      }) as JwtPayload;

      return decoded;
    } catch (error) {
      if (error instanceof jwt.JsonWebTokenError) {
        auditLogger.logSecurityEvent('INVALID_ACCESS_TOKEN', null, false, { 
          error: error.name,
          message: 'Token verification failed' 
        });
      }
      return null;
    }
  }

  getRefreshTokenExpiry(): Date {
    const expiry = new Date();
    expiry.setDate(expiry.getDate() + 7); // 7 days
    return expiry;
  }

  isTokenExpired(expiresAt: Date): boolean {
    return new Date() > expiresAt;
  }
}

export const tokenService = new TokenService();

import bcrypt from 'bcrypt';
import { Pool } from 'pg';
import { User, PublicUser, RefreshToken, UserRole } from '../models/user';
import { tokenService } from './tokenService';
import { auditLogger } from '../utils/auditLogger';

export class AuthService {
  constructor(private db: Pool) {}

  private toPublicUser(user: User): PublicUser {
    return {
      id: user.id,
      email: user.email,
      role: user.role,
      last_login: user.last_login
    };
  }

  private async timingSafeEquals(a: string, b: string): Promise<boolean> {
    // Use bcrypt.compare for timing-safe string comparison
    const hashA = await bcrypt.hash(a, 1);
    const hashB = await bcrypt.hash(b, 1);
    return bcrypt.compare(a, hashA) && bcrypt.compare(b, hashB);
  }

  async authenticateUser(email: string, password: string, ipAddress?: string): Promise<{ user: PublicUser; accessToken: string; refreshToken: string } | null> {
    const client = await this.db.connect();
    
    try {
      // Always perform timing-safe operations even for non-existent users
      const userQuery = 'SELECT * FROM users WHERE email = $1';
      const userResult = await client.query(userQuery, [email]);
      
      const user = userResult.rows[0] as User | undefined;
      
      // Check if account is locked
      if (user && user.locked_until && user.locked_until > new Date()) {
        auditLogger.logAuthAttempt(email, 'LOGIN_BLOCKED_LOCKED', ipAddress, false, { reason: 'Account locked' });
        return null;
      }

      // Verify password (always hash even if user doesn't exist for timing safety)
      const isValidPassword = user ? 
        await bcrypt.compare(password, user.password_hash) :
        await bcrypt.compare(password, '$2b$12$dummy.hash.for.timing.safety.purposes.only');

      if (!user || !isValidPassword) {
        // Increment failed login attempts if user exists
        if (user) {
          const newFailedAttempts = user.failed_login_attempts + 1;
          const lockUntil = newFailedAttempts >= 5 ? new Date(Date.now() + 15 * 60 * 1000) : null; // Lock for 15 minutes after 5 failed attempts
          
          await client.query(
            'UPDATE users SET failed_login_attempts = $1, locked_until = $2 WHERE id = $3',
            [newFailedAttempts, lockUntil, user.id]
          );
        }

        auditLogger.logAuthAttempt(email, 'LOGIN_FAILED', ipAddress, false, { reason: 'Invalid credentials' });
        return null;
      }

      // Reset failed login attempts on successful login
      await client.query(
        'UPDATE users SET failed_login_attempts = 0, locked_until = NULL, last_login = CURRENT_TIMESTAMP WHERE id = $1',
        [user.id]
      );

      // Generate tokens
      const accessToken = tokenService.generateAccessToken(user.id, user.email, user.role);
      const refreshToken = tokenService.generateRefreshToken();
      const refreshTokenHash = await tokenService.hashRefreshToken(refreshToken);

      // Store refresh token in database
      await client.query(
        'INSERT INTO refresh_tokens (user_id, token_hash, expires_at) VALUES ($1, $2, $3)',
        [user.id, refreshTokenHash, tokenService.getRefreshTokenExpiry()]
      );

      auditLogger.logAuthAttempt(email, 'LOGIN_SUCCESS', ipAddress, true, { userId: user.id });

      return {
        user: this.toPublicUser(user),
        accessToken,
        refreshToken
      };
    } finally {
      client.release();
    }
  }

  async refreshAccessToken(refreshToken: string, ipAddress?: string): Promise<{ accessToken: string; refreshToken: string } | null> {
    const client = await this.db.connect();

    try {
      // Find refresh token in database
      const tokenQuery = `
        SELECT rt.*, u.id as user_id, u.email, u.role 
        FROM refresh_tokens rt
        JOIN users u ON rt.user_id = u.id
        WHERE rt.revoked_at IS NULL AND rt.expires_at > CURRENT_TIMESTAMP
      `;
      const tokenResult = await client.query(tokenQuery);

      // Find matching token using timing-safe comparison
      let matchedToken: RefreshToken & { user_id: number; email: string; role: UserRole } | null = null;
      
      for (const dbToken of tokenResult.rows) {
        const isMatch = await tokenService.validateRefreshToken(refreshToken, dbToken.token_hash);
        if (isMatch) {
          matchedToken = dbToken;
          break;
        }
      }

      if (!matchedToken || tokenService.isTokenExpired(matchedToken.expires_at)) {
        auditLogger.logSecurityEvent('REFRESH_TOKEN_INVALID', null, false, { ipAddress });
        return null;
      }

      // Check for token reuse (security breach)
      if (matchedToken.replaced_by_token) {
        // Token has already been used - possible replay attack
        await client.query('UPDATE refresh_tokens SET revoked_at = CURRENT_TIMESTAMP WHERE user_id = $1', [matchedToken.user_id]);
        auditLogger.logSecurityEvent('REFRESH_TOKEN_REUSE_DETECTED', matchedToken.email, false, { 
          userId: matchedToken.user_id, 
          ipAddress 
        });
        return null;
      }

      // Generate new tokens
      const newAccessToken = tokenService.generateAccessToken(matchedToken.user_id, matchedToken.email, matchedToken.role);
      const newRefreshToken = tokenService.generateRefreshToken();
      const newRefreshTokenHash = await tokenService.hashRefreshToken(newRefreshToken);

      // Revoke old refresh token and create new one (token rotation)
      await client.query(
        'UPDATE refresh_tokens SET revoked_at = CURRENT_TIMESTAMP, replaced_by_token = $1 WHERE id = $2',
        [newRefreshToken, matchedToken.id]
      );

      await client.query(
        'INSERT INTO refresh_tokens (user_id, token_hash, expires_at) VALUES ($1, $2, $3)',
        [matchedToken.user_id, newRefreshTokenHash, tokenService.getRefreshTokenExpiry()]
      );

      auditLogger.logAuthAttempt(matchedToken.email, 'TOKEN_REFRESH_SUCCESS', ipAddress, true, { 
        userId: matchedToken.user_id 
      });

      return {
        accessToken: newAccessToken,
        refreshToken: newRefreshToken
      };
    } finally {
      client.release();
    }
  }

  async revokeRefreshToken(refreshToken: string, ipAddress?: string): Promise<boolean> {
    const client = await this.db.connect();

    try {
      const tokenQuery = 'SELECT * FROM refresh_tokens WHERE revoked_at IS NULL';
      const tokenResult = await client.query(tokenQuery);

      // Find and revoke matching token
      for (const dbToken of tokenResult.rows) {
        const isMatch = await tokenService.validateRefreshToken(refreshToken, dbToken.token_hash);
        if (isMatch) {
          await client.query(
            'UPDATE refresh_tokens SET revoked_at = CURRENT_TIMESTAMP WHERE id = $1',
            [dbToken.id]
          );
          
          auditLogger.logAuthAttempt(null, 'LOGOUT_SUCCESS', ipAddress, true, { 
            userId: dbToken.user_id 
          });
          return true;
        }
      }

      return false;
    } finally {
      client.release();
    }
  }

  async revokeAllUserTokens(userId: number): Promise<void> {
    const client = await this.db.connect();
    
    try {
      await client.query(
        'UPDATE refresh_tokens SET revoked_at = CURRENT_TIMESTAMP WHERE user_id = $1 AND revoked_at IS NULL',
        [userId]
      );
    } finally {
      client.release();
    }
  }
}
