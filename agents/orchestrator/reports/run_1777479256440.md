```typescript
import React from 'react';
import { DndContext, DragEndEvent, DragOverlay, DragStartEvent } from '@dnd-kit/core';
import { useState } from 'react';
import { JobCard } from './JobCard';
import { DroppableColumn } from './DroppableColumn';
import { useJobDragDrop } from '../hooks/useJobDragDrop';
import { Job } from '../types/job';

interface WeeklyViewProps {
  jobs: Job[];
  onJobUpdate: (jobId: string, job: Job) => void;
  weekStartDate: Date;
}

export const WeeklyView: React.FC<WeeklyViewProps> = ({
  jobs,
  onJobUpdate,
  weekStartDate
}) => {
  const [activeJob, setActiveJob] = useState<Job | null>(null);
  const { handleDragEnd, isUpdating } = useJobDragDrop(onJobUpdate);

  const handleDragStart = (event: DragStartEvent) => {
    const job = jobs.find(j => j.id === event.active.id);
    setActiveJob(job || null);
  };

  const handleDragEndInternal = (event: DragEndEvent) => {
    setActiveJob(null);
    handleDragEnd(event);
  };

  const getDayDate = (dayOffset: number): Date => {
    const date = new Date(weekStartDate);
    date.setDate(date.getDate() + dayOffset);
    return date;
  };

  const getJobsForDay = (dayOffset: number): Job[] => {
    const dayDate = getDayDate(dayOffset);
    return jobs.filter(job => {
      if (!job.scheduledStart) return false;
      const jobDate = new Date(job.scheduledStart);
      return jobDate.toDateString() === dayDate.toDateString();
    });
  };

  const unscheduledJobs = jobs.filter(job => !job.scheduledStart);

  return (
    <DndContext onDragStart={handleDragStart} onDragEnd={handleDragEndInternal}>
      <div className="weekly-view" data-updating={isUpdating}>
        <div className="week-columns">
          {[0, 1, 2, 3, 4, 5, 6].map(dayOffset => (
            <DroppableColumn
              key={dayOffset}
              id={`day-${dayOffset}`}
              title={getDayDate(dayOffset).toLocaleDateString('en-US', { weekday: 'short', month: 'short', day: 'numeric' })}
              jobs={getJobsForDay(dayOffset)}
            />
          ))}
        </div>
        
        <DroppableColumn
          id="unscheduled"
          title="Unscheduled"
          jobs={unscheduledJobs}
          className="unscheduled-pool"
        />
      </div>

      <DragOverlay>
        {activeJob && (
          <div className="drag-overlay">
            <div className="job-title">{activeJob.title}</div>
          </div>
        )}
      </DragOverlay>
    </DndContext>
  );
};
```

### 2. Custom Hook for Drag Logic

**File: `hooks/useJobDragDrop.ts`** (New file)
```typescript
import { useState, useCallback } from 'react';
import { DragEndEvent } from '@dnd-kit/core';
import { updateJob } from '../api/jobs';
import { Job } from '../types/job';

interface UseJobDragDropReturn {
  handleDragEnd: (event: DragEndEvent) => Promise<void>;
  isUpdating: boolean;
}

export const useJobDragDrop = (
  onJobUpdate: (jobId: string, job: Job) => void
): UseJobDragDropReturn => {
  const [isUpdating, setIsUpdating] = useState(false);

  const calculateScheduledStart = (dropZoneId: string, originalJob: Job): Date | null => {
    if (dropZoneId === 'unscheduled') {
      return null;
    }

    const dayMatch = dropZoneId.match(/^day-(\d+)$/);
    if (!dayMatch) {
      return originalJob.scheduledStart;
    }

    const dayOffset = parseInt(dayMatch[1], 10);
    const weekStart = getWeekStart(); // This should come from WeeklyView context
    const targetDate = new Date(weekStart);
    targetDate.setDate(targetDate.getDate() + dayOffset);

    // Preserve existing time if job already has scheduledStart, otherwise default to 08:00
    if (originalJob.scheduledStart) {
      const existingTime = new Date(originalJob.scheduledStart);
      targetDate.setHours(existingTime.getHours(), existingTime.getMinutes(), 0, 0);
    } else {
      targetDate.setHours(8, 0, 0, 0); // Default to 08:00
    }

    return targetDate;
  };

  const handleDragEnd = useCallback(async (event: DragEndEvent) => {
    const { active, over } = event;
    
    if (!over || !active.data.current) {
      return;
    }

    const job = active.data.current as Job;
    const dropZoneId = over.id as string;
    
    const newScheduledStart = calculateScheduledStart(dropZoneId, job);
    
    // Skip if no change needed
    if (newScheduledStart?.getTime() === job.scheduledStart?.getTime()) {
      return;
    }

    const optimisticJob: Job = {
      ...job,
      scheduledStart: newScheduledStart
    };

    // Optimistic update
    onJobUpdate(job.id, optimisticJob);
    setIsUpdating(true);

    try {
      const updatedJob = await updateJob(job.id, { scheduledStart: newScheduledStart });
      onJobUpdate(job.id, updatedJob);
    } catch (error) {
      // Rollback on failure
      onJobUpdate(job.id, job);
      console.error('Failed to update job schedule:', error);
      // TODO: Show user notification
    } finally {
      setIsUpdating(false);
    }
  }, [onJobUpdate]);

  return {
    handleDragEnd,
    isUpdating
  };
};

// Helper function - should be moved to utils or context
const getWeekStart = (): Date => {
  const now = new Date();
  const day = now.getDay();
  const diff = now.getDate() - day;
  return new Date(now.setDate(diff));
};
```

```typescript
import { Job } from '../types/job';

const API_BASE = process.env.NEXT_PUBLIC_API_BASE || '/api';

interface UpdateJobPayload {
  scheduledStart?: Date | null;
  // other job fields...
}

export const updateJob = async (jobId: string, payload: UpdateJobPayload): Promise<Job> => {
  if (!jobId || typeof jobId !== 'string') {
    throw new Error('Invalid job ID provided');
  }

  const response = await fetch(`${API_BASE}/jobs/${encodeURIComponent(jobId)}`, {
    method: 'PATCH',
    headers: {
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      ...payload,
      scheduledStart: payload.scheduledStart?.toISOString() ?? null
    }),
  });

  if (!response.ok) {
    const errorText = await response.text();
    throw new Error(`Failed to update job: ${response.status} ${errorText}`);
  }

  const updatedJob = await response.json();
  
  // Validate response structure
  if (!updatedJob || typeof updatedJob.id !== 'string') {
    throw new Error('Invalid response from server');
  }

  return {
    ...updatedJob,
    scheduledStart: updatedJob.scheduledStart ? new Date(updatedJob.scheduledStart) : null
  };
};
```
