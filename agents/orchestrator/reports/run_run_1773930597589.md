import { PrismaClient } from '@prisma/client';
import { CreateJobRequest, UpdateJobRequest, JobQuery, JobStatus } from '../types/job.types.js';

const prisma = new PrismaClient();

export class JobService {
  private getBaseWhereClause(userRole: string, userId: string, isDeleted: boolean = false) {
    const baseWhere: any = {
      deletedAt: isDeleted ? { not: null } : null
    };

    // Drivers can only see their own assigned jobs
    if (userRole === 'DRIVER') {
      baseWhere.assignedDriverId = userId;
    }

    return baseWhere;
  }

  private validateStatusTransition(currentStatus: JobStatus, newStatus: JobStatus): boolean {
    const validTransitions: Record<JobStatus, JobStatus[]> = {
      [JobStatus.DRAFT]: [JobStatus.ASSIGNED],
      [JobStatus.ASSIGNED]: [JobStatus.DRAFT, JobStatus.IN_PROGRESS],
      [JobStatus.IN_PROGRESS]: [JobStatus.COMPLETED],
      [JobStatus.COMPLETED]: [] // No transitions from completed
    };

    return validTransitions[currentStatus].includes(newStatus);
  }

  async createJob(data: CreateJobRequest, createdById: string) {
    return await prisma.job.create({
      data: {
        ...data,
        createdById,
        scheduledAt: data.scheduledAt ? new Date(data.scheduledAt) : null
      },
      include: {
        createdBy: {
          select: { id: true, name: true, email: true }
        },
        assignedDriver: {
          select: { id: true, name: true, email: true }
        }
      }
    });
  }

  async getJobs(query: JobQuery, userRole: string, userId: string) {
    const { status, assignedDriverId, page = 1, limit = 10 } = query;
    
    const whereClause = {
      ...this.getBaseWhereClause(userRole, userId),
      ...(status && { status }),
      ...(assignedDriverId && userRole !== 'DRIVER' && { assignedDriverId })
    };

    const [jobs, total] = await Promise.all([
      prisma.job.findMany({
        where: whereClause,
        include: {
          createdBy: {
            select: { id: true, name: true, email: true }
          },
          assignedDriver: {
            select: { id: true, name: true, email: true }
          }
        },
        orderBy: { createdAt: 'desc' },
        skip: (page - 1) * limit,
        take: limit
      }),
      prisma.job.count({ where: whereClause })
    ]);

    return {
      jobs,
      pagination: {
        page,
        limit,
        total,
        pages: Math.ceil(total / limit)
      }
    };
  }

  async getJobById(id: string, userRole: string, userId: string) {
    const whereClause = {
      id,
      ...this.getBaseWhereClause(userRole, userId)
    };

    const job = await prisma.job.findFirst({
      where: whereClause,
      include: {
        createdBy: {
          select: { id: true, name: true, email: true }
        },
        assignedDriver: {
          select: { id: true, name: true, email: true }
        }
      }
    });

    return job;
  }

  async updateJob(id: string, data: UpdateJobRequest, userRole: string, userId: string) {
    // First check if job exists and user has access
    const existingJob = await this.getJobById(id, userRole, userId);
    if (!existingJob) {
      return null;
    }

    // Validate status transition if status is being updated
    if (data.status && data.status !== existingJob.status) {
      if (!this.validateStatusTransition(existingJob.status as JobStatus, data.status)) {
        throw new Error(`Invalid status transition from ${existingJob.status} to ${data.status}`);
      }
    }

    const updateData: any = { ...data };
    
    // Handle scheduledAt conversion
    if (data.scheduledAt !== undefined) {
      updateData.scheduledAt = data.scheduledAt ? new Date(data.scheduledAt) : null;
    }

    return await prisma.job.update({
      where: { id },
      data: updateData,
      include: {
        createdBy: {
          select: { id: true, name: true, email: true }
        },
        assignedDriver: {
          select: { id: true, name: true, email: true }
        }
      }
    });
  }

  async deleteJob(id: string) {
    // Check if job exists and is not already deleted
    const existingJob = await prisma.job.findFirst({
      where: { id, deletedAt: null }
    });

    if (!existingJob) {
      return null;
    }

    return await prisma.job.update({
      where: { id },
      data: { deletedAt: new Date() }
    });
  }
}

import { Request, Response } from 'express';
import { JobService } from '../services/job.service.js';
import { createJobSchema, updateJobSchema, jobQuerySchema } from '../types/job.types.js';

const jobService = new JobService();

interface AuthenticatedRequest extends Request {
  user?: {
    id: string;
    email: string;
    role: string;
  };
}

export const createJob = async (req: AuthenticatedRequest, res: Response) => {
  try {
    const validation = createJobSchema.safeParse(req.body);
    if (!validation.success) {
      return res.status(400).json({
        error: 'Validation failed',
        details: validation.error.errors
      });
    }

    const job = await jobService.createJob(validation.data, req.user!.id);
    
    res.status(201).json({ data: job });
  } catch (error) {
    console.error('Create job error:', error);
    res.status(500).json({ error: 'Internal server error' });
  }
};

export const getJobs = async (req: AuthenticatedRequest, res: Response) => {
  try {
    const validation = jobQuerySchema.safeParse(req.query);
    if (!validation.success) {
      return res.status(400).json({
        error: 'Invalid query parameters',
        details: validation.error.errors
      });
    }

    const result = await jobService.getJobs(
      validation.data,
      req.user!.role,
      req.user!.id
    );
    
    res.json({ data: result });
  } catch (error) {
    console.error('Get jobs error:', error);
    res.status(500).json({ error: 'Internal server error' });
  }
};

export const getJobById = async (req: AuthenticatedRequest, res: Response) => {
  try {
    const { id } = req.params;
    
    const job = await jobService.getJobById(id, req.user!.role, req.user!.id);
    
    if (!job) {
      return res.status(404).json({ error: 'Job not found' });
    }
    
    res.json({ data: job });
  } catch (error) {
    console.error('Get job by ID error:', error);
    res.status(500).json({ error: 'Internal server error' });
  }
};

export const updateJob = async (req: AuthenticatedRequest, res: Response) => {
  try {
    const { id } = req.params;
    
    const validation = updateJobSchema.safeParse(req.body);
    if (!validation.success) {
      return res.status(400).json({
        error: 'Validation failed',
        details: validation.error.errors
      });
    }

    const job = await jobService.updateJob(
      id,
      validation.data,
      req.user!.role,
      req.user!.id
    );
    
    if (!job) {
      return res.status(404).json({ error: 'Job not found' });
    }
    
    res.json({ data: job });
  } catch (error) {
    if (error instanceof Error && error.message.includes('Invalid status transition')) {
      return res.status(400).json({ error: error.message });
    }
    
    console.error('Update job error:', error);
    res.status(500).json({ error: 'Internal server error' });
  }
};

export const deleteJob = async (req: AuthenticatedRequest, res: Response) => {
  try {
    const { id } = req.params;
    
    const job = await jobService.deleteJob(id);
    
    if (!job) {
      return res.status(404).json({ error: 'Job not found' });
    }
    
    res.json({ data: { message: 'Job deleted successfully' } });
  } catch (error) {
    console.error('Delete job error:', error);
    res.status(500).json({ error: 'Internal server error' });
  }
};

import { Router } from 'express';
import { authenticateToken } from '../middleware/authenticate.js';
import { requireRole } from '../middleware/requireRole.js';
import {
  createJob,
  getJobs,
  getJobById,
  updateJob,
  deleteJob
} from '../controllers/job.controller.js';

const router = Router();

// All routes require authentication
router.use(authenticateToken);

// POST /api/jobs - Admin and Dispatcher only
router.post('/', requireRole(['ADMIN', 'DISPATCHER']), createJob);

// GET /api/jobs - All roles (filtered by role in service)
router.get('/', getJobs);

// GET /api/jobs/:id - All roles (filtered by role in service)
router.get('/:id', getJobById);

// PATCH /api/jobs/:id - Admin and Dispatcher only
router.patch('/:id', requireRole(['ADMIN', 'DISPATCHER']), updateJob);

// DELETE /api/jobs/:id - Admin only
router.delete('/:id', requireRole(['ADMIN']), deleteJob);

export default router;
