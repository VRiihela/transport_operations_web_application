```typescript
import { Request, Response, NextFunction } from 'express';
import { jobService } from '../services/jobService';

export const validateJobOwnership = async (req: Request, res: Response, next: NextFunction): Promise<void> => {
  try {
    const jobId = req.params.id;
    const userId = req.user?.id; // Assumes user is attached by requireAuth middleware
    
    if (!userId) {
      res.status(401).json({ error: 'Authentication required' });
      return;
    }

    const job = await jobService.getJobById(jobId);
    
    if (!job) {
      res.status(404).json({ error: 'Job not found' });
      return;
    }

    if (job.assignedDriverId !== userId) {
      res.status(403).json({ error: 'Only the assigned driver can update job notes' });
      return;
    }

    req.job = job; // Attach job to request for controller use
    next();
  } catch (error) {
    console.error('Job ownership validation error:', error);
    res.status(500).json({ error: 'Internal server error' });
  }
};
```

```typescript
import React, { useState, useCallback } from 'react';
import { Job, JobStatus } from '../types/job';
import { updateJobNotes } from '../api/jobsApi';
import { useDebounce } from '../hooks/useDebounce'; // Assumes existing debounce hook

interface JobNotesProps {
  job: Job;
  onNotesUpdate: (jobId: string, notes: string) => void;
}

const JobNotes: React.FC<JobNotesProps> = ({ job, onNotesUpdate }) => {
  const [notes, setNotes] = useState(job.driverNotes || '');
  const [isSaving, setIsSaving] = useState(false);
  const [saveStatus, setSaveStatus] = useState<'idle' | 'saved' | 'error'>('idle');
  
  const debouncedNotes = useDebounce(notes, 1000); // Auto-save after 1s of no typing
  
  const isEditable = job.status === JobStatus.ASSIGNED || job.status === JobStatus.IN_PROGRESS;

  const handleSave = useCallback(async (notesToSave: string) => {
    if (notesToSave === (job.driverNotes || '')) return; // No changes
    
    setIsSaving(true);
    setSaveStatus('idle');
    
    try {
      await updateJobNotes(job.id, notesToSave);
      onNotesUpdate(job.id, notesToSave);
      setSaveStatus('saved');
      setTimeout(() => setSaveStatus('idle'), 2000);
    } catch (error) {
      console.error('Failed to save notes:', error);
      setSaveStatus('error');
      setTimeout(() => setSaveStatus('idle'), 5000);
    } finally {
      setIsSaving(false);
    }
  }, [job.id, job.driverNotes, onNotesUpdate]);

  // Auto-save when debounced notes change
  React.useEffect(() => {
    if (isEditable && debouncedNotes !== (job.driverNotes || '')) {
      handleSave(debouncedNotes);
    }
  }, [debouncedNotes, handleSave, isEditable, job.driverNotes]);

  return (
    <div className="job-notes">
      <label htmlFor={`notes-${job.id}`} className="notes-label">
        Driver Notes:
      </label>
      <textarea
        id={`notes-${job.id}`}
        value={notes}
        onChange={(e) => setNotes(e.target.value)}
        disabled={!isEditable || isSaving}
        readOnly={!isEditable}
        maxLength={1000}
        rows={3}
        placeholder={isEditable ? "Add notes or deviation info..." : "No notes"}
        className={`notes-textarea ${!isEditable ? 'read-only' : ''}`}
      />
      <div className="notes-status">
        <span className="char-count">{notes.length}/1000</span>
        {isSaving && <span className="saving">Saving...</span>}
        {saveStatus === 'saved' && <span className="saved">Saved ✓</span>}
        {saveStatus === 'error' && <span className="error">Save failed - please try again</span>}
      </div>
    </div>
  );
};

// In the main MyJobsPage component, add JobNotes to each job row:
const MyJobsPage: React.FC = () => {
  const [jobs, setJobs] = useState<Job[]>([]);
  
  const handleNotesUpdate = (jobId: string, newNotes: string) => {
    setJobs(prev => prev.map(job => 
      job.id === jobId ? { ...job, driverNotes: newNotes } : job
    ));
  };

  return (
    <div className="my-jobs-page">
      {/* Existing job list rendering */}
      {jobs.map(job => (
        <div key={job.id} className="job-row">
          {/* Existing job display elements */}
          <JobNotes job={job} onNotesUpdate={handleNotesUpdate} />
        </div>
      ))}
    </div>
  );
};
```
