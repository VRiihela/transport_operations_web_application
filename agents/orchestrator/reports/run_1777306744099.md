import { Request, Response } from 'express';
import { User } from '../models/User';
import { EditUserPayload, EditUserResponse } from '../types/user';
import { validateEmail, sanitizeString } from '../utils/validation';

export const editUser = async (req: Request, res: Response): Promise<void> => {
  try {
    const userId = req.params.id;
    const payload: EditUserPayload = req.body;

    // Validate user exists
    const existingUser = await User.findById(userId);
    if (!existingUser) {
      res.status(404).json({
        success: false,
        error: 'User not found'
      } as EditUserResponse);
      return;
    }

    // Validate and sanitize inputs
    const updateData: Partial<User> = {};

    if (payload.name !== undefined) {
      const sanitizedName = sanitizeString(payload.name);
      if (!sanitizedName || sanitizedName.length < 1 || sanitizedName.length > 100) {
        res.status(400).json({
          success: false,
          error: 'Name must be between 1 and 100 characters'
        } as EditUserResponse);
        return;
      }
      updateData.name = sanitizedName;
    }

    if (payload.email !== undefined) {
      const sanitizedEmail = sanitizeString(payload.email).toLowerCase();
      if (!validateEmail(sanitizedEmail)) {
        res.status(400).json({
          success: false,
          error: 'Invalid email format'
        } as EditUserResponse);
        return;
      }

      // Check email uniqueness (excluding current user)
      const emailExists = await User.findOne({ 
        email: sanitizedEmail, 
        id: { $ne: userId } 
      });
      if (emailExists) {
        res.status(400).json({
          success: false,
          error: 'Email already in use'
        } as EditUserResponse);
        return;
      }
      updateData.email = sanitizedEmail;
    }

    if (payload.role !== undefined) {
      const validRoles = ['admin', 'user', 'moderator'];
      if (!validRoles.includes(payload.role)) {
        res.status(400).json({
          success: false,
          error: 'Invalid role specified'
        } as EditUserResponse);
        return;
      }
      updateData.role = payload.role;
    }

    if (payload.isActive !== undefined) {
      if (typeof payload.isActive !== 'boolean') {
        res.status(400).json({
          success: false,
          error: 'isActive must be a boolean'
        } as EditUserResponse);
        return;
      }
      updateData.isActive = payload.isActive;
    }

    // Update user
    const updatedUser = await User.findByIdAndUpdate(
      userId,
      { ...updateData, updatedAt: new Date() },
      { new: true, select: '-password' }
    );

    res.status(200).json({
      success: true,
      user: updatedUser
    } as EditUserResponse);

  } catch (error) {
    console.error('Error editing user:', error);
    res.status(500).json({
      success: false,
      error: 'Internal server error'
    } as EditUserResponse);
  }
};

export const validateEmail = (email: string): boolean => {
  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  return emailRegex.test(email) && email.length <= 254;
};

export const sanitizeString = (input: string): string => {
  if (typeof input !== 'string') return '';
  return input.trim().replace(/[<>'"]/g, '');
};

import { Request, Response, NextFunction } from 'express';
import jwt from 'jsonwebtoken';

interface AuthenticatedRequest extends Request {
  user?: {
    id: string;
    role: string;
  };
}

export const requireAdmin = (
  req: AuthenticatedRequest, 
  res: Response, 
  next: NextFunction
): void => {
  try {
    const token = req.headers.authorization?.split(' ')[1];
    
    if (!token) {
      res.status(401).json({ success: false, error: 'No token provided' });
      return;
    }

    const decoded = jwt.verify(token, process.env.JWT_SECRET!) as any;
    req.user = decoded;

    if (req.user.role !== 'admin') {
      res.status(403).json({ success: false, error: 'Admin access required' });
      return;
    }

    next();
  } catch (error) {
    res.status(401).json({ success: false, error: 'Invalid token' });
  }
};

import { User, EditUserPayload } from '../types/user';

class UserService {
  private baseUrl = '/api/users';

  private async makeRequest<T>(
    url: string, 
    options: RequestInit
  ): Promise<T> {
    const token = localStorage.getItem('authToken');
    const response = await fetch(url, {
      ...options,
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${token}`,
        ...options.headers,
      },
    });

    if (!response.ok) {
      const error = await response.json();
      throw new Error(error.error || 'Request failed');
    }

    return response.json();
  }

  async editUser(
    userId: string, 
    payload: EditUserPayload
  ): Promise<{ success: boolean; user?: User; error?: string }> {
    return this.makeRequest(`${this.baseUrl}/${userId}`, {
      method: 'PATCH',
      body: JSON.stringify(payload),
    });
  }
}

export const userService = new UserService();
