import { PrismaClient, JobStatus } from '@prisma/client';
import { AppError } from '../utils/AppError';

const prisma = new PrismaClient();

export class TeamService {
  static async addMemberWithJobReassignment(teamId: string, driverId: string) {
    return await prisma.$transaction(async (tx) => {
      // Verify team exists and is active
      const team = await tx.teams.findFirst({
        where: { id: teamId, isActive: true }
      });
      
      if (!team) {
        throw new AppError('Team not found or inactive', 404);
      }

      // Verify driver exists
      const driver = await tx.users.findFirst({
        where: { id: driverId, role: 'DRIVER' }
      });

      if (!driver) {
        throw new AppError('Driver not found', 404);
      }

      // Check if driver is already in any active team
      const existingMembership = await tx.teamMembers.findFirst({
        where: {
          driverId,
          team: { isActive: true }
        }
      });

      if (existingMembership) {
        throw new AppError('Driver is already a member of an active team', 400);
      }

      // Add driver to team
      await tx.teamMembers.create({
        data: {
          teamId,
          driverId,
        }
      });

      // Reassign driver's ASSIGNED/IN_PROGRESS jobs to the team
      const reassignedJobs = await tx.jobs.updateMany({
        where: {
          assignedDriverId: driverId,
          status: { in: [JobStatus.ASSIGNED, JobStatus.IN_PROGRESS] }
        },
        data: {
          assignedDriverId: null,
          assignedTeamId: teamId,
        }
      });

      return {
        success: true,
        reassignedJobsCount: reassignedJobs.count,
        teamId,
        driverId,
      };
    });
  }

  static async removeMember(teamId: string, driverId: string) {
    return await prisma.$transaction(async (tx) => {
      // Verify membership exists
      const membership = await tx.teamMembers.findFirst({
        where: { teamId, driverId }
      });

      if (!membership) {
        throw new AppError('Driver is not a member of this team', 404);
      }

      // Remove team member
      await tx.teamMembers.delete({
        where: {
          teamId_driverId: {
            teamId,
            driverId
          }
        }
      });

      return { success: true };
    });
  }

  static async dissolveTeamWithValidation(teamId: string) {
    return await prisma.$transaction(async (tx) => {
      // Check for active team jobs
      const activeJobs = await tx.jobs.findMany({
        where: {
          assignedTeamId: teamId,
          status: { in: [JobStatus.ASSIGNED, JobStatus.IN_PROGRESS] }
        },
        select: { id: true, status: true }
      });

      if (activeJobs.length > 0) {
        throw new AppError('Cannot dissolve team with active jobs', 409);
      }

      // Set all team jobs to DRAFT and clear team assignment
      await tx.jobs.updateMany({
        where: { assignedTeamId: teamId },
        data: {
          status: JobStatus.DRAFT,
          assignedTeamId: null,
        }
      });

      // Remove all team members
      await tx.teamMembers.deleteMany({
        where: { teamId }
      });

      // Delete team
      await tx.teams.delete({
        where: { id: teamId }
      });

      return { success: true };
    });
  }

  static async getAvailableDrivers() {
    // Get drivers who are not members of any active team
    const availableDrivers = await prisma.users.findMany({
      where: {
        role: 'DRIVER',
        teamMembers: {
          none: {
            team: {
              isActive: true
            }
          }
        }
      },
      select: {
        id: true,
        name: true,
        email: true,
      }
    });

    return availableDrivers;
  }

  static async getActiveTeams() {
    return await prisma.teams.findMany({
      where: { isActive: true },
      select: {
        id: true,
        name: true,
        members: {
          select: {
            driver: {
              select: {
                id: true,
                name: true,
              }
            }
          }
        }
      }
    });
  }
}

export const useAvailableDrivers = () => {
  const [drivers, setDrivers] = useState<AvailableDriver[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const fetchDrivers = async () => {
    try {
      setLoading(true);
      const response = await fetch('/api/teams/available-drivers', {
        headers: {
          'Authorization': `Bearer ${localStorage.getItem('token')}`,
        },
      });

      if (!response.ok) {
        throw new Error('Failed to fetch available drivers');
      }

      const data = await response.json();
      setDrivers(data.data);
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Unknown error');
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchDrivers();
  }, []);

  return { drivers, loading, error, refetch: fetchDrivers };
};
