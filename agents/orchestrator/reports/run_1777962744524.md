import { Request, Response, NextFunction } from 'express';

// Assuming User type exists with role property
interface AuthenticatedRequest extends Request {
  user?: {
    id: string;
    role: string;
    // other user properties
  };
}

export const adminOrDispatcher = (
  req: AuthenticatedRequest,
  res: Response,
  next: NextFunction
): void => {
  if (!req.user) {
    res.status(401).json({ error: 'Authentication required' });
    return;
  }

  if (req.user.role !== 'ADMIN' && req.user.role !== 'DISPATCHER') {
    res.status(403).json({ error: 'Admin or Dispatcher role required' });
    return;
  }

  next();
};

```typescript
import { Router, Request, Response } from 'express';
import { PrismaClient } from '@prisma/client';
import { 
  CreateCustomerSchema, 
  CustomerSearchQuerySchema, 
  CustomerParamsSchema,
  type CreateCustomerData,
  type CustomerSearchQuery,
  type CustomerParams
} from '../models/customer';
import { authenticateToken, adminOrDispatcher } from '../middleware/auth';

const router = Router();
const prisma = new PrismaClient();

// Apply authentication and authorization to all routes
router.use(authenticateToken, adminOrDispatcher);

// GET /api/customers?phone=<partial> - Search customers by phone
router.get('/', async (req: Request, res: Response): Promise<void> => {
  try {
    const query = CustomerSearchQuerySchema.parse(req.query);
    
    const customers = await prisma.customer.findMany({
      where: {
        phone: {
          contains: query.phone,
          mode: 'insensitive',
        },
      },
      select: {
        id: true,
        name: true,
        phone: true,
        email: true,
        type: true,
        companyName: true,
      },
      take: 10,
      orderBy: {
        createdAt: 'desc',
      },
    });

    res.json(customers);
  } catch (error) {
    if (error instanceof Error && 'issues' in error) {
      res.status(400).json({ error: 'Invalid query parameters', details: error.issues });
      return;
    }
    
    console.error('Customer search error:', error);
    res.status(500).json({ error: 'Internal server error' });
  }
});

// POST /api/customers - Create new customer
router.post('/', async (req: Request, res: Response): Promise<void> => {
  try {
    const customerData: CreateCustomerData = CreateCustomerSchema.parse(req.body);
    
    const customer = await prisma.customer.create({
      data: {
        name: customerData.name,
        phone: customerData.phone,
        email: customerData.email || null,
        type: customerData.type,
        companyName: customerData.companyName || null,
        referenceNumber: customerData.referenceNumber || null,
      },
      select: {
        id: true,
        name: true,
        phone: true,
        email: true,
        type: true,
        companyName: true,
        referenceNumber: true,
        createdAt: true,
      },
    });

    res.status(201).json(customer);
  } catch (error) {
    if (error instanceof Error && 'issues' in error) {
      res.status(400).json({ error: 'Invalid request body', details: error.issues });
      return;
    }
    
    // Handle Prisma unique constraint violation
    if (error && typeof error === 'object' && 'code' in error && error.code === 'P2002') {
      res.status(409).json({ error: 'Customer with this phone number already exists' });
      return;
    }
    
    console.error('Customer creation error:', error);
    res.status(500).json({ error: 'Internal server error' });
  }
});

// GET /api/customers/:id - Get single customer
router.get('/:id', async (req: Request, res: Response): Promise<void> => {
  try {
    const params: CustomerParams = CustomerParamsSchema.parse(req.params);
    
    const customer = await prisma.customer.findUnique({
      where: { id: params.id },
      select: {
        id: true,
        name: true,
        phone: true,
        email: true,
        type: true,
        companyName: true,
        referenceNumber: true,
        createdAt: true,
      },
    });

    if (!customer) {
      res.status(404).json({ error: 'Customer not found' });
      return;
    }

    res.json(customer);
  } catch (error) {
    if (error instanceof Error && 'issues' in error) {
      res.status(400).json({ error: 'Invalid customer ID', details: error.issues });
      return;
    }
    
    console.error('Customer fetch error:', error);
    res.status(500).json({ error: 'Internal server error' });
  }
});

export default router;
```
