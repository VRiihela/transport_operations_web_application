### 4. Completion Report Service

**File:** `src/services/completion-report-service.ts`
```typescript
import { PrismaClient } from '@prisma/client';
import { CreateCompletionReportRequest, CompletionReport } from '../types/completion-report';

const prisma = new PrismaClient();

export class CompletionReportService {
  static calculateTotalHours(actualStart: Date, actualEnd: Date): number {
    const diffMs = actualEnd.getTime() - actualStart.getTime();
    return Math.round((diffMs / (1000 * 60 * 60)) * 100) / 100; // Round to 2 decimal places
  }

  static async upsertCompletionReport(
    jobId: string,
    userId: string,
    data: CreateCompletionReportRequest
  ): Promise<CompletionReport> {
    const actualStart = new Date(data.actualStart);
    const actualEnd = new Date(data.actualEnd);
    const totalHours = this.calculateTotalHours(actualStart, actualEnd);

    const completionReport = await prisma.completionReport.upsert({
      where: { jobId },
      create: {
        jobId,
        workDescription: data.workDescription,
        actualStart,
        actualEnd,
        totalHours,
        customerName: data.customerName,
        customerSignature: data.customerSignature,
        createdById: userId
      },
      update: {
        workDescription: data.workDescription,
        actualStart,
        actualEnd,
        totalHours,
        customerName: data.customerName,
        customerSignature: data.customerSignature,
        approvedAt: null // Reset approval when updating
      }
    });

    return completionReport;
  }

  static async approveCompletionReport(jobId: string): Promise<CompletionReport> {
    const existingReport = await prisma.completionReport.findUnique({
      where: { jobId }
    });

    if (!existingReport) {
      throw new Error('Completion report not found');
    }

    if (!existingReport.customerSignature || existingReport.customerSignature.trim() === '') {
      throw new Error('Customer signature is required for approval');
    }

    const approvedReport = await prisma.completionReport.update({
      where: { jobId },
      data: {
        approvedAt: new Date()
      }
    });

    return approvedReport;
  }

  static async getCompletionReport(jobId: string): Promise<CompletionReport | null> {
    return await prisma.completionReport.findUnique({
      where: { jobId }
    });
  }

  static async hasApprovedCompletionReport(jobId: string): Promise<boolean> {
    const report = await prisma.completionReport.findUnique({
      where: { jobId },
      select: { approvedAt: true }
    });

    return report !== null && report.approvedAt !== null;
  }
}
```

```typescript
import { PrismaClient } from '@prisma/client';

const prisma = new PrismaClient();

export class CompletionReportAuth {
  static async canModifyCompletionReport(
    userId: string, 
    userRole: string, 
    jobId: string
  ): Promise<boolean> {
    // Admin and Dispatcher can modify any completion report
    if (userRole === 'Admin' || userRole === 'Dispatcher') {
      return true;
    }

    // Driver can only modify if they are assigned to the job
    if (userRole === 'Driver') {
      const job = await prisma.job.findUnique({
        where: { id: jobId },
        select: { assignedDriverId: true }
      });

      return job?.assignedDriverId === userId;
    }

    return false;
  }

  static async validateJobExists(jobId: string): Promise<boolean> {
    const job = await prisma.job.findUnique({
      where: { id: jobId },
      select: { id: true }
    });

    return job !== null;
  }
}
```

```typescript
import express, { Request, Response } from 'express';
import { authenticateToken } from '../middleware/auth';
import { 
  validateCreateCompletionReport, 
  validateApproveCompletionReport 
} from '../middleware/completion-report-validation';
import { CompletionReportService } from '../services/completion-report-service';
import { CompletionReportAuth } from '../services/completion-report-auth';

const router = express.Router();

// POST /api/jobs/:id/completion-report - Create/Update completion report
router.post(
  '/jobs/:id/completion-report',
  authenticateToken,
  validateCreateCompletionReport,
  async (req: Request, res: Response) => {
    try {
      const jobId = req.params.id;
      const userId = req.user?.id;
      const userRole = req.user?.role;

      if (!userId || !userRole) {
        return res.status(401).json({ error: 'Authentication required' });
      }

      // Validate job exists
      const jobExists = await CompletionReportAuth.validateJobExists(jobId);
      if (!jobExists) {
        return res.status(404).json({ error: 'Job not found' });
      }

      // Check authorization
      const canModify = await CompletionReportAuth.canModifyCompletionReport(
        userId, 
        userRole, 
        jobId
      );

      if (!canModify) {
        return res.status(403).json({ 
          error: 'Not authorized to modify completion report for this job' 
        });
      }

      const completionReport = await CompletionReportService.upsertCompletionReport(
        jobId,
        userId,
        req.body
      );

      res.status(200).json({ data: completionReport });
    } catch (error) {
      console.error('Error creating/updating completion report:', error);
      res.status(500).json({ error: 'Internal server error' });
    }
  }
);

// POST /api/jobs/:id/completion-report/approve - Approve completion report
router.post(
  '/jobs/:id/completion-report/approve',
  authenticateToken,
  validateApproveCompletionReport,
  async (req: Request, res: Response) => {
    try {
      const jobId = req.params.id;
      const userId = req.user?.id;
      const userRole = req.user?.role;

      if (!userId || !userRole) {
        return res.status(401).json({ error: 'Authentication required' });
      }

      // Check authorization
      const canModify = await CompletionReportAuth.canModifyCompletionReport(
        userId, 
        userRole, 
        jobId
      );

      if (!canModify) {
        return res.status(403).json({ 
          error: 'Not authorized to approve completion report for this job' 
        });
      }

      const approvedReport = await CompletionReportService.approveCompletionReport(jobId);
      res.status(200).json({ data: approvedReport });

    } catch (error) {
      if (error instanceof Error) {
        if (error.message === 'Completion report not found') {
          return res.status(404).json({ error: error.message });
        }
        if (error.message === 'Customer signature is required for approval') {
          return res.status(400).json({ error: error.message });
        }
      }
      
      console.error('Error approving completion report:', error);
      res.status(500).json({ error: 'Internal server error' });
    }
  }
);

export default router;
```
