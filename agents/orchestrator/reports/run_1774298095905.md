export const validateCompletionReport = [
  param('id').isUUID().withMessage('Invalid job ID'),
  body('workDescription')
    .trim()
    .isLength({ min: 10, max: 2000 })
    .withMessage('Work description must be between 10 and 2000 characters'),
  body('actualStart')
    .isISO8601()
    .withMessage('Invalid actual start datetime'),
  body('actualEnd')
    .isISO8601()
    .withMessage('Invalid actual end datetime')
    .custom((value, { req }) => {
      const start = new Date(req.body.actualStart);
      const end = new Date(value);
      if (end <= start) {
        throw new Error('Actual end must be after actual start');
      }
      return true;
    }),
  handleValidationErrors
];

import { CompletionReport, CreateCompletionReportRequest, ApproveCompletionReportRequest } from '../types/job';

export const createCompletionReport = async (req: Request, res: Response): Promise<void> => {
  try {
    const jobId = req.params.id;
    const userId = req.user?.id;
    const { workDescription, actualStart, actualEnd }: CreateCompletionReportRequest = req.body;

    // Verify job exists and user has access
    const job = await Job.findByPk(jobId);
    if (!job) {
      res.status(404).json({ error: 'Job not found' });
      return;
    }

    if (job.assignedTo !== userId && req.user?.role !== 'admin') {
      res.status(403).json({ error: 'Unauthorized' });
      return;
    }

    if (job.status !== 'IN_PROGRESS') {
      res.status(400).json({ error: 'Job must be in progress to create completion report' });
      return;
    }

    // Create completion report
    const completionReport = await CompletionReport.create({
      jobId,
      workDescription: workDescription.trim(),
      actualStart: new Date(actualStart),
      actualEnd: new Date(actualEnd)
    });

    // Update job with completion report reference
    await job.update({ completionReportId: completionReport.id });

    res.status(201).json(completionReport);
  } catch (error) {
    console.error('Error creating completion report:', error);
    res.status(500).json({ error: 'Internal server error' });
  }
};

export const approveCompletionReport = async (req: Request, res: Response): Promise<void> => {
  try {
    const jobId = req.params.id;
    const userId = req.user?.id;
    const { customerName, signatureData }: ApproveCompletionReportRequest = req.body;

    // Verify job and completion report exist
    const job = await Job.findByPk(jobId, {
      include: [{ model: CompletionReport, as: 'completionReport' }]
    });

    if (!job) {
      res.status(404).json({ error: 'Job not found' });
      return;
    }

    if (job.assignedTo !== userId && req.user?.role !== 'admin') {
      res.status(403).json({ error: 'Unauthorized' });
      return;
    }

    if (!job.completionReport) {
      res.status(400).json({ error: 'Completion report not found' });
      return;
    }

    if (job.completionReport.approvedAt) {
      res.status(400).json({ error: 'Completion report already approved' });
      return;
    }

    // Validate signature is not blank (basic check - actual blank detection is client-side)
    if (!signatureData || signatureData.length < 100) {
      res.status(400).json({ error: 'Invalid signature data' });
      return;
    }

    // Update completion report with approval
    await job.completionReport.update({
      customerName: customerName.trim(),
      signatureData,
      approvedAt: new Date()
    });

    res.json({ success: true });
  } catch (error) {
    console.error('Error approving completion report:', error);
    res.status(500).json({ error: 'Internal server error' });
  }
};

import React, { useRef, useEffect, useState, useCallback } from 'react';

interface SignatureCanvasProps {
  width?: number;
  height?: number;
  onSignatureChange: (hasSignature: boolean, signatureData?: string) => void;
}

interface Point {
  x: number;
  y: number;
}

export const SignatureCanvas: React.FC<SignatureCanvasProps> = ({ 
  width = 400, 
  height = 200, 
  onSignatureChange 
}) => {
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const [isDrawing, setIsDrawing] = useState(false);
  const [hasDrawn, setHasDrawn] = useState(false);

  const getCanvasPoint = useCallback((event: MouseEvent | TouchEvent): Point | null => {
    const canvas = canvasRef.current;
    if (!canvas) return null;

    const rect = canvas.getBoundingClientRect();
    const scaleX = canvas.width / rect.width;
    const scaleY = canvas.height / rect.height;

    if ('touches' in event) {
      if (event.touches.length === 0) return null;
      return {
        x: (event.touches[0].clientX - rect.left) * scaleX,
        y: (event.touches[0].clientY - rect.top) * scaleY
      };
    } else {
      return {
        x: (event.clientX - rect.left) * scaleX,
        y: (event.clientY - rect.top) * scaleY
      };
    }
  }, []);

  const startDrawing = useCallback((event: MouseEvent | TouchEvent) => {
    event.preventDefault();
    const canvas = canvasRef.current;
    const ctx = canvas?.getContext('2d');
    if (!canvas || !ctx) return;

    const point = getCanvasPoint(event);
    if (!point) return;

    setIsDrawing(true);
    ctx.beginPath();
    ctx.moveTo(point.x, point.y);
  }, [getCanvasPoint]);

  const draw = useCallback((event: MouseEvent | TouchEvent) => {
    event.preventDefault();
    if (!isDrawing) return;

    const canvas = canvasRef.current;
    const ctx = canvas?.getContext('2d');
    if (!canvas || !ctx) return;

    const point = getCanvasPoint(event);
    if (!point) return;

    ctx.lineTo(point.x, point.y);
    ctx.stroke();
    
    if (!hasDrawn) {
      setHasDrawn(true);
      onSignatureChange(true);
    }
  }, [isDrawing, hasDrawn, getCanvasPoint, onSignatureChange]);

  const stopDrawing = useCallback(() => {
    setIsDrawing(false);
  }, []);

  const clearCanvas = useCallback(() => {
    const canvas = canvasRef.current;
    const ctx = canvas?.getContext('2d');
    if (!canvas || !ctx) return;

    ctx.clearRect(0, 0, canvas.width, canvas.height);
    setHasDrawn(false);
    onSignatureChange(false);
  }, [onSignatureChange]);

  const getSignatureData = useCallback((): string | null => {
    const canvas = canvasRef.current;
    if (!canvas || !hasDrawn) return null;
    return canvas.toDataURL('image/png');
  }, [hasDrawn]);

  // Expose getSignatureData method to parent
  useEffect(() => {
    const canvas = canvasRef.current;
    if (canvas) {
      (canvas as any).getSignatureData = getSignatureData;
    }
  }, [getSignatureData]);

  useEffect(() => {
    const canvas = canvasRef.current;
    const ctx = canvas?.getContext('2d');
    if (!canvas || !ctx) return;

    // Set up canvas drawing style
    ctx.strokeStyle = '#000000';
    ctx.lineWidth = 2;
    ctx.lineCap = 'round';
    ctx.lineJoin = 'round';

    // Mouse events
    canvas.addEventListener('mousedown', startDrawing);
    canvas.addEventListener('mousemove', draw);
    canvas.addEventListener('mouseup', stopDrawing);
    canvas.addEventListener('mouseout', stopDrawing);

    // Touch events
    canvas.addEventListener('touchstart', startDrawing, { passive: false });
    canvas.addEventListener('touchmove', draw, { passive: false });
    canvas.addEventListener('touchend', stopDrawing);

    return () => {
      canvas.removeEventListener('mousedown', startDrawing);
      canvas.removeEventListener('mousemove', draw);
      canvas.removeEventListener('mouseup', stopDrawing);
      canvas.removeEventListener('mouseout', stopDrawing);
      canvas.removeEventListener('touchstart', startDrawing);
      canvas.removeEventListener('touchmove', draw);
      canvas.removeEventListener('touchend', stopDrawing);
    };
  }, [startDrawing, draw, stopDrawing]);

  return (
    <div className="signature-canvas-container">
      <canvas
        ref={canvasRef}
        width={width}
        height={height}
        className="border border-gray-300 bg-white cursor-crosshair touch-none"
        style={{ touchAction: 'none' }}
      />
      <div className="mt-2">
        <button
          type="button"
          onClick={clearCanvas}
          className="px-3 py-1 text-sm bg-gray-100 hover:bg-gray-200 border rounded"
        >
          Clear
        </button>
      </div>
    </div>
  );
};

export const formatFinnishDateTime = (dateString: string): string => {
  const date = new Date(dateString);
  return new Intl.DateTimeFormat('fi-FI', {
    year: 'numeric',
    month: '2-digit',
    day: '2-digit',
    hour: '2-digit',
    minute: '2-digit',
    timeZone: 'Europe/Helsinki'
  }).format(date);
};

export const formatFinnishDateTimeRange = (start: string, end: string): string => {
  const startFormatted = formatFinnishDateTime(start);
  const endFormatted = formatFinnishDateTime(end);
  
  // If same date, show date once
  const startDate = startFormatted.split(' ')[0];
  const endDate = endFormatted.split(' ')[0];
  
  if (startDate === endDate) {
    const startTime = startFormatted.split(' ')[1];
    const endTime = endFormatted.split(' ')[1];
    return `${startDate} ${startTime}–${endTime}`;
  }
  
  return `${startFormatted}–${endFormatted}`;
};

export const calculateHours = (start: string, end: string): number => {
  const startDate = new Date(start);
  const endDate = new Date(end);
  const diffMs = endDate.getTime() - startDate.getTime();
  return Math.max(0, diffMs / (1000 * 60 * 60)); // Convert ms to hours
};
