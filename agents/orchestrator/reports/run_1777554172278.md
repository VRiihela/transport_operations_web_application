```typescript
import { format, parse, isValid, startOfDay, endOfDay } from 'date-fns';
import { fi } from 'date-fns/locale';

export const formatDateFinnish = (date: Date): string => {
  return format(date, 'dd.MM.yyyy', { locale: fi });
};

export const formatDateISO = (date: Date): string => {
  return format(date, 'yyyy-MM-dd');
};

export const parseFinnishDate = (dateString: string): Date | null => {
  try {
    const parsed = parse(dateString, 'dd.MM.yyyy', new Date());
    return isValid(parsed) ? parsed : null;
  } catch {
    return null;
  }
};

export const isJobScheduledForDate = (job: Job, selectedDate: Date): boolean => {
  if (!job.scheduledStart) return false;
  
  const jobDate = new Date(job.scheduledStart);
  const selectedStart = startOfDay(selectedDate);
  const selectedEnd = endOfDay(selectedDate);
  
  return jobDate >= selectedStart && jobDate <= selectedEnd;
};

export const sortJobsByScheduledTime = (jobs: Job[]): Job[] => {
  return [...jobs].sort((a, b) => {
    if (!a.scheduledStart && !b.scheduledStart) return 0;
    if (!a.scheduledStart) return 1;
    if (!b.scheduledStart) return -1;
    
    return new Date(a.scheduledStart).getTime() - new Date(b.scheduledStart).getTime();
  });
};
```

```typescript
import { useState, useEffect, useCallback } from 'react';
import { Job, JobsApiParams, JobsApiResponse } from '../types/job.types';
import { fetchJobs } from '../api/jobs.api';
import { formatDateISO } from '../utils/dateUtils';

interface UseJobsResult {
  jobs: Job[];
  loading: boolean;
  error: string | null;
  refetch: (params?: JobsApiParams) => Promise<void>;
}

export const useJobs = (initialParams?: JobsApiParams): UseJobsResult => {
  const [jobs, setJobs] = useState<Job[]>([]);
  const [loading, setLoading] = useState<boolean>(false);
  const [error, setError] = useState<string | null>(null);

  const fetchJobsData = useCallback(async (params: JobsApiParams = {}): Promise<void> => {
    setLoading(true);
    setError(null);

    try {
      const response: JobsApiResponse = await fetchJobs(params);
      setJobs(response.jobs);
    } catch (err) {
      const errorMessage = err instanceof Error ? err.message : 'Failed to fetch jobs';
      setError(errorMessage);
      console.error('Error fetching jobs:', err);
    } finally {
      setLoading(false);
    }
  }, []);

  const refetch = useCallback(async (params?: JobsApiParams): Promise<void> => {
    await fetchJobsData(params || initialParams || {});
  }, [fetchJobsData, initialParams]);

  useEffect(() => {
    fetchJobsData(initialParams || {});
  }, [fetchJobsData, initialParams]);

  return {
    jobs,
    loading,
    error,
    refetch
  };
};

export const useJobsForDate = (selectedDate: Date): UseJobsResult => {
  const dateParam = formatDateISO(selectedDate);
  
  return useJobs({
    date: dateParam
  });
};
```

### 5. src/api/jobs.api.ts
```typescript
import { JobsApiParams, JobsApiResponse } from '../types/job.types';

const API_BASE_URL = process.env.REACT_APP_API_BASE_URL || '/api';

export const fetchJobs = async (params: JobsApiParams = {}): Promise<JobsApiResponse> => {
  const searchParams = new URLSearchParams();
  
  // Validate and sanitize parameters
  if (params.date) {
    // Validate ISO date format (YYYY-MM-DD)
    const dateRegex = /^\d{4}-\d{2}-\d{2}$/;
    if (!dateRegex.test(params.date)) {
      throw new Error('Invalid date format. Expected YYYY-MM-DD');
    }
    searchParams.append('date', params.date);
  }
  
  if (params.driverId) {
    // Basic validation for driverId
    if (typeof params.driverId !== 'string' || params.driverId.trim().length === 0) {
      throw new Error('Invalid driverId parameter');
    }
    searchParams.append('driverId', params.driverId.trim());
  }
  
  if (params.status) {
    const validStatuses = ['unassigned', 'assigned', 'in_progress', 'completed'];
    if (!validStatuses.includes(params.status)) {
      throw new Error('Invalid status parameter');
    }
    searchParams.append('status', params.status);
  }

  const url = `${API_BASE_URL}/jobs?${searchParams.toString()}`;
  
  try {
    const response = await fetch(url, {
      method: 'GET',
      headers: {
        'Content-Type': 'application/json',
        // Add authentication headers if needed
        ...(process.env.REACT_APP_API_KEY && {
          'X-API-Key': process.env.REACT_APP_API_KEY
        })
      }
    });

    if (!response.ok) {
      if (response.status === 400) {
        const errorData = await response.json().catch(() => ({}));
        throw new Error(errorData.message || 'Bad request - invalid parameters');
      }
      throw new Error(`HTTP ${response.status}: ${response.statusText}`);
    }

    const data: JobsApiResponse = await response.json();
    
    // Validate response structure
    if (!data || !Array.isArray(data.jobs)) {
      throw new Error('Invalid response format from jobs API');
    }

    return data;
  } catch (error) {
    if (error instanceof Error) {
      throw error;
    }
    throw new Error('Network error while fetching jobs');
  }
};
```

```typescript
import React, { useState, useMemo, useEffect } from 'react';
import { DatePicker } from './DatePicker';
import { useJobsForDate } from '../../hooks/useJobs';
import { Job } from '../../types/job.types';
import { isJobScheduledForDate, sortJobsByScheduledTime } from '../../utils/dateUtils';

interface Driver {
  id: string;
  name: string;
}

interface AssignViewProps {
  drivers: Driver[];
}

export const AssignView: React.FC<AssignViewProps> = ({ drivers }) => {
  const [selectedDate, setSelectedDate] = useState<Date>(new Date());
  const { jobs, loading, error, refetch } = useJobsForDate(selectedDate);

  // Refetch jobs when date changes
  useEffect(() => {
    refetch();
  }, [selectedDate, refetch]);

  const { driverJobs, unassignedJobs } = useMemo(() => {
    const driverJobsMap = new Map<string, Job[]>();
    const unassigned: Job[] = [];

    // Initialize driver job arrays
    drivers.forEach(driver => {
      driverJobsMap.set(driver.id, []);
    });

    jobs.forEach(job => {
      if (!job.driverId || !driverJobsMap.has(job.driverId)) {
        // Job is unassigned OR assigned to non-existent driver
        // Show in unassigned pool if: no scheduledStart OR matches selected date
        if (!job.scheduledStart || isJobScheduledForDate(job, selectedDate)) {
          unassigned.push(job);
        }
      } else {
        // Job is assigned to a valid driver
        // Show in driver column only if scheduledStart matches selected date
        if (job.scheduledStart && isJobScheduledForDate(job, selectedDate)) {
          driverJobsMap.get(job.driverId)!.push(job);
        }
      }
    });

    // Sort jobs within each driver column by scheduled time
    driverJobsMap.forEach((driverJobs, driverId) => {
      driverJobsMap.set(driverId, sortJobsByScheduledTime(driverJobs));
    });

    // Sort unassigned jobs by scheduled time
    const sortedUnassigned = sortJobsByScheduledTime(unassigned);

    return {
      driverJobs: driverJobsMap,
      unassignedJobs: sortedUnassigned
    };
  }, [jobs, drivers, selectedDate]);

  const handleDateChange = (newDate: Date): void => {
    setSelectedDate(newDate);
  };

  if (error) {
    return (
      <div className="assign-view-error" role="alert">
        <h3>Virhe ladatessa töitä</h3>
        <p>{error}</p>
        <button onClick={() => refetch()} className="retry-button">
          Yritä uudelleen
        </button>
      </div>
    );
  }

  return (
    <div className="assign-view">
      <header className="assign-view-header">
        <h2>Työn jako</h2>
        <DatePicker
          selectedDate={selectedDate}
          onDateChange={handleDateChange}
          className="assign-date-picker"
        />
      </header>

      <div className="assign-view-content">
        {loading && (
          <div className="assign-view-loading" aria-live="polite">
            Ladataan töitä...
          </div>
        )}

        <div className="assign-columns">
          {/* Unassigned pool */}
          <div className="assign-column unassigned-column">
            <h3>Jakamatta ({unassignedJobs.length})</h3>
            <div className="job-list">
              {unassignedJobs.map(job => (
                <div key={job.id} className="job-card unassigned" data-job-id={job.id}>
                  <h4>{job.title}</h4>
                  {job.scheduledStart && (
                    <div className="job-time">
                      {new Date(job.scheduledStart).toLocaleTimeString('fi-FI', {
                        hour: '2-digit',
                        minute: '2-digit'
                      })}
                    </div>
                  )}
                </div>
              ))}
            </div>
          </div>

          {/* Driver columns */}
          {drivers.map(driver => {
            const driverJobList = driverJobs.get(driver.id) || [];
            
            return (
              <div key={driver.id} className="assign-column driver-column">
                <h3>{driver.name} ({driverJobList.length})</h3>
                <div className="job-list">
                  {driverJobList.map(job => (
                    <div key={job.id} className="job-card assigned" data-job-id={job.id}>
                      <h4>{job.title}</h4>
                      {job.scheduledStart && (
                        <div className="job-time">
                          {new Date(job.scheduledStart).toLocaleTimeString('fi-FI', {
                            hour: '2-digit',
                            minute: '2-digit'
                          })}
                        </div>
                      )}
                    </div>
                  ))}
                </div>
              </div>
            );
          })}
        </div>
      </div>
    </div>
  );
};
```
