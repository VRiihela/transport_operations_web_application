import { z } from 'zod';

const addressValidation = {
  street: z.string().min(1, 'Street is required').trim(),
  houseNumber: z.string().min(1, 'House number is required').trim(),
  stair: z.string().optional(),
  postalCode: z.string().min(1, 'Postal code is required').trim(),
  city: z.string().min(1, 'City is required').trim(),
};

const deliveryAddressValidation = {
  deliveryStreet: z.string().optional(),
  deliveryHouseNumber: z.string().optional(),
  deliveryStair: z.string().optional(),
  deliveryPostalCode: z.string().optional(),
  deliveryCity: z.string().optional(),
};

export const createJobSchema = z
  .object({
    title: z.string().min(1, 'Title is required').trim(),
    description: z.string().optional(),
    ...addressValidation,
    ...deliveryAddressValidation,
  })
  .refine(
    (data) => {
      const deliveryFields = [
        data.deliveryStreet,
        data.deliveryHouseNumber,
        data.deliveryPostalCode,
        data.deliveryCity,
      ];
      
      const hasAnyDelivery = deliveryFields.some(field => field && field.trim().length > 0);
      
      if (!hasAnyDelivery) return true;
      
      // If any delivery field is provided, required fields must be present
      return deliveryFields.every(field => field && field.trim().length > 0);
    },
    {
      message: 'If delivery address is provided, street, house number, postal code, and city are all required',
      path: ['deliveryStreet'],
    }
  );

export const updateJobSchema = z
  .object({
    title: z.string().min(1).trim().optional(),
    description: z.string().optional(),
    street: z.string().min(1).trim().optional(),
    houseNumber: z.string().min(1).trim().optional(),
    stair: z.string().optional(),
    postalCode: z.string().min(1).trim().optional(),
    city: z.string().min(1).trim().optional(),
    ...deliveryAddressValidation,
  })
  .refine(
    (data) => {
      const deliveryFields = [
        data.deliveryStreet,
        data.deliveryHouseNumber,
        data.deliveryPostalCode,
        data.deliveryCity,
      ];
      
      const hasAnyDelivery = deliveryFields.some(field => field && field.trim().length > 0);
      
      if (!hasAnyDelivery) return true;
      
      // If any delivery field is provided, required fields must be present
      return deliveryFields.every(field => field && field.trim().length > 0);
    },
    {
      message: 'If delivery address is provided, street, house number, postal code, and city are all required',
      path: ['deliveryStreet'],
    }
  );

export type CreateJobData = z.infer<typeof createJobSchema>;

import { NextApiRequest, NextApiResponse } from 'next';
import { updateJobSchema } from '@/lib/validations/job';
import { prisma } from '@/lib/prisma';
import { Job } from '@/types/job';

interface ApiError {
  error: string;
  details?: string[];
}

export default async function handler(
  req: NextApiRequest,
  res: NextApiResponse<Job | ApiError>
) {
  const { id } = req.query;

  if (typeof id !== 'string') {
    return res.status(400).json({ error: 'Invalid job ID' });
  }

  try {
    if (req.method === 'PATCH') {
      const validation = updateJobSchema.safeParse(req.body);
      
      if (!validation.success) {
        return res.status(400).json({
          error: 'Validation failed',
          details: validation.error.errors.map(err => `${err.path.join('.')}: ${err.message}`),
        });
      }

      const updateData = validation.data;

      // Convert undefined to null for optional fields
      const processedData = Object.fromEntries(
        Object.entries(updateData).map(([key, value]) => [
          key,
          value === undefined ? undefined : (value === '' ? null : value),
        ])
      );

      const job = await prisma.job.update({
        where: { id },
        data: processedData,
      });

      return res.status(200).json({
        ...job,
        createdAt: job.createdAt.toISOString(),
        updatedAt: job.updatedAt.toISOString(),
      });
    }

    return res.status(405).json({ error: 'Method not allowed' });
  } catch (error) {
    console.error('API Error:', error);
    
    if ((error as any)?.code === 'P2025') {
      return res.status(404).json({ error: 'Job not found' });
    }
    
    return res.status(500).json({ error: 'Internal server error' });
  }
}
