import { Router } from 'express';
import { z } from 'zod';
import { requireAuth, requireRole } from '../middleware/auth';
import { Job } from '../models/Job';
import { updateJobSchema } from '../schemas/jobSchemas';
import { AppError } from '../utils/errors';

const router = Router();

// PATCH /api/jobs/:id - Update job (Admin/Dispatcher only)
router.patch('/:id', requireAuth, requireRole(['Admin', 'Dispatcher']), async (req, res, next) => {
  try {
    const jobId = req.params.id;
    const userId = req.user.id;
    
    // Validate request body
    const validatedData = updateJobSchema.parse(req.body);
    
    // Find existing job (exclude soft-deleted)
    const existingJob = await Job.findOne({
      where: { id: jobId, deletedAt: null }
    });
    
    if (!existingJob) {
      throw new AppError('Job not found', 404);
    }
    
    // Merge with existing values for business rule validation
    const mergedData = {
      ...existingJob.dataValues,
      ...validatedData
    };
    
    // Business rule: schedulingNote required if both scheduled times are null
    if (!mergedData.scheduledStart && !mergedData.scheduledEnd && !mergedData.schedulingNote) {
      throw new AppError('Scheduling note is required when no scheduled times are provided', 400);
    }
    
    // Update job (explicitly exclude createdById)
    const { createdById, ...updateData } = validatedData;
    const [updatedRowsCount, updatedJobs] = await Job.update(updateData, {
      where: { id: jobId, deletedAt: null },
      returning: true
    });
    
    if (updatedRowsCount === 0) {
      throw new AppError('Job not found', 404);
    }
    
    const updatedJob = updatedJobs[0];
    
    res.json({
      data: updatedJob
    });
    
  } catch (error) {
    if (error instanceof z.ZodError) {
      return res.status(400).json({
        error: 'Validation error',
        details: error.errors
      });
    }
    next(error);
  }
});

export default router;

import { z } from 'zod';

// Existing createJobSchema (assumed to exist)
export const createJobSchema = z.object({
  title: z.string().min(1, 'Title is required').max(200),
  description: z.string().optional(),
  scheduledStart: z.string().datetime().nullable().optional(),
  scheduledEnd: z.string().datetime().nullable().optional(),
  schedulingNote: z.string().optional(),
  assignedDriverId: z.string().uuid().nullable().optional(),
  createdById: z.string().uuid()
});

// New update schema - partial fields, excluding createdById
export const updateJobSchema = createJobSchema
  .omit({ createdById: true })
  .partial()
  .refine((data) => {
    // Allow empty updates
    return Object.keys(data).length > 0;
  }, {
    message: 'At least one field must be provided for update'
  });

  isOpen,
  onClose,
  onSuccess

  const [formData, setFormData] = useState<FormData>({
    title: '',
    description: '',
    scheduledStart: '',
    scheduledEnd: '',
    schedulingNote: '',
    assignedDriverId: ''
  });
  
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [error, setError] = useState<string | null>(null);

  // Pre-fill form when job changes
  useEffect(() => {
    if (job) {
      setFormData({
        title: job.title || '',
        description: job.description || '',
        scheduledStart: job.scheduledStart ? 
          new Date(job.scheduledStart).toISOString().slice(0, 16) : '',
        scheduledEnd: job.scheduledEnd ? 
          new Date(job.scheduledEnd).toISOString().slice(0, 16) : '',
        schedulingNote: job.schedulingNote || '',
        assignedDriverId: job.assignedDriverId || ''
      });
      setError(null);
    }
  }, [job]);

  const handleInputChange = (e: React.ChangeEvent<HTMLInputElement | HTMLTextAreaElement | HTMLSelectElement>) => {
    const { name, value } = e.target;
    setFormData(prev => ({
      ...prev,
      [name]: value
    }));
  };

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setIsSubmitting(true);
    setError(null);

    try {
      // Prepare payload with only changed fields and proper types
      const payload: Partial<Job> = {};
      
      if (formData.title !== job.title) payload.title = formData.title;
      if (formData.description !== job.description) payload.description = formData.description;
      if (formData.schedulingNote !== job.schedulingNote) payload.schedulingNote = formData.schedulingNote;

        return;

      onClose();

      setError(errorMessage);
    } finally {
      setIsSubmitting(false);
    }
  };

      <div className="modal-content" onClick={(e) => e.stopPropagation()}>

import { Job } from '../types/job.types';

const API_BASE = '/api';

export const updateJob = async (jobId: string, data: Partial<Job>): Promise<Job> => {
  const response = await fetch(`${API_BASE}/jobs/${jobId}`, {
    method: 'PATCH',
    headers: {
      'Content-Type': 'application/json',
    },
    credentials: 'include',
    body: JSON.stringify(data)
  });

  if (!response.ok) {
    const error = await response.json().catch(() => ({ error: 'Failed to update job' }));
    throw new Error(error.error || `HTTP ${response.status}`);
  }

  const result = await response.json();
  return result.data;
};
