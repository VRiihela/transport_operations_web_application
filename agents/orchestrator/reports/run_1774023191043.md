```typescript
import React, { useState, useEffect } from 'react';
import { useAuth } from '../context/AuthContext';
import { axiosInstance } from '../api/axios';
import styles from './UsersPage.module.css';

interface User {
  id: string;
  email: string;
  role: 'ADMIN' | 'DISPATCHER' | 'DRIVER';
  isActive: boolean;
}

interface CreateUserForm {
  email: string;
  password: string;
  role: 'DISPATCHER' | 'DRIVER';
}

interface ResetPasswordForm {
  newPassword: string;
}

const UsersPage: React.FC = () => {
  const { user } = useAuth();
  const [users, setUsers] = useState<User[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  const [showCreateForm, setShowCreateForm] = useState(false);
  const [resetPasswordUser, setResetPasswordUser] = useState<User | null>(null);

  const [createForm, setCreateForm] = useState<CreateUserForm>({
    email: '',
    password: '',
    role: 'DRIVER'
  });

  const [resetForm, setResetForm] = useState<ResetPasswordForm>({
    newPassword: ''
  });

  const [submitting, setSubmitting] = useState(false);

  useEffect(() => {
    fetchUsers();
  }, []);

  const fetchUsers = async (): Promise<void> => {
    try {
      setLoading(true);
      const response = await axiosInstance.get<User[]>('/api/users');
      setUsers(response.data);
      setError(null);
    } catch (err) {
      setError('Failed to fetch users. Please try again.');
      console.error('Error fetching users:', err);
    } finally {
      setLoading(false);
    }
  };

  const handleCreateUser = async (e: React.FormEvent): Promise<void> => {
    e.preventDefault();
    if (submitting) return;

    setSubmitting(true);
    setError(null);

    try {
      const response = await axiosInstance.post<User>('/api/users', createForm);
      setUsers(prev => [...prev, response.data]);
      setCreateForm({ email: '', password: '', role: 'DRIVER' });
      setShowCreateForm(false);
    } catch (err: any) {
      if (err.response?.status === 409) {
        setError('Email already exists. Please use a different email.');
      } else {
        setError('Failed to create user. Please try again.');
      }
      console.error('Error creating user:', err);
    } finally {
      setSubmitting(false);
    }
  };

  const handleToggleActive = async (userId: string, currentStatus: boolean): Promise<void> => {
    if (userId === user?.id) return; // Prevent admin from deactivating themselves

    try {
      await axiosInstance.patch(`/api/users/${userId}`, {
        isActive: !currentStatus
      });
      
      setUsers(prev => 
        prev.map(u => 
          u.id === userId ? { ...u, isActive: !currentStatus } : u
        )
      );
    } catch (err) {
      setError('Failed to update user status. Please try again.');
      console.error('Error toggling user status:', err);
    }
  };

  const handleResetPassword = async (e: React.FormEvent): Promise<void> => {
    e.preventDefault();
    if (submitting || !resetPasswordUser) return;

    setSubmitting(true);
    setError(null);

    try {
      await axiosInstance.post(`/api/users/${resetPasswordUser.id}/reset-password`, {
        newPassword: resetForm.newPassword
      });
      
      setResetForm({ newPassword: '' });
      setResetPasswordUser(null);
      // Show success message briefly
      const successMsg = 'Password reset successfully';
      setError(null);
      // Could add a success state here if needed
    } catch (err) {
      setError('Failed to reset password. Please try again.');
      console.error('Error resetting password:', err);
    } finally {
      setSubmitting(false);
    }
  };

  const getRoleBadgeClass = (role: User['role']): string => {
    switch (role) {
      case 'ADMIN': return styles.roleAdmin;
      case 'DISPATCHER': return styles.roleDispatcher;
      case 'DRIVER': return styles.roleDriver;
      default: return '';
    }
  };

  if (loading) {
    return <div className={styles.container}>Loading users...</div>;
  }

  return (
    <div className={styles.container}>
      <div className={styles.header}>
        <h1>User Management</h1>
        <button 
          className={styles.createButton}
          onClick={() => setShowCreateForm(true)}
        >
          Create User
        </button>
      </div>

      {error && <div className={styles.error}>{error}</div>}

      <div className={styles.tableContainer}>
        <table className={styles.table}>
          <thead>
            <tr>
              <th>Email</th>
              <th>Role</th>
              <th>Status</th>
              <th>Actions</th>
            </tr>
          </thead>
          <tbody>
            {users.map(userItem => (
              <tr 
                key={userItem.id}
                className={!userItem.isActive ? styles.inactiveRow : ''}
              >
                <td className={!userItem.isActive ? styles.strikethrough : ''}>
                  {userItem.email}
                </td>
                <td>
                  <span className={`${styles.roleBadge} ${getRoleBadgeClass(userItem.role)}`}>
                    {userItem.role}
                  </span>
                </td>
                <td>
                  <span className={userItem.isActive ? styles.activeStatus : styles.inactiveStatus}>
                    {userItem.isActive ? 'Active' : 'Inactive'}
                  </span>
                </td>
                <td>
                  <div className={styles.actions}>
                    {userItem.id !== user?.id && (
                      <button
                        className={userItem.isActive ? styles.deactivateButton : styles.activateButton}
                        onClick={() => handleToggleActive(userItem.id, userItem.isActive)}
                      >
                        {userItem.isActive ? 'Deactivate' : 'Activate'}
                      </button>
                    )}
                    <button
                      className={styles.resetButton}
                      onClick={() => setResetPasswordUser(userItem)}
                    >
                      Reset Password
                    </button>
                  </div>
                </td>
              </tr>
            ))}
          </tbody>
        </table>
      </div>

      {/* Create User Modal */}
      {showCreateForm && (
        <div className={styles.modal}>
          <div className={styles.modalContent}>
            <h2>Create New User</h2>
            <form onSubmit={handleCreateUser}>
              <div className={styles.formGroup}>
                <label htmlFor="email">Email *</label>
                <input
                  id="email"
                  type="email"
                  required
                  value={createForm.email}
                  onChange={(e) => setCreateForm(prev => ({
                    ...prev,
                    email: e.target.value
                  }))}
                />
              </div>
              
              <div className={styles.formGroup}>
                <label htmlFor="password">Password *</label>
                <input
                  id="password"
                  type="password"
                  required
                  value={createForm.password}
                  onChange={(e) => setCreateForm(prev => ({
                    ...prev,
                    password: e.target.value
                  }))}
                />
              </div>
              
              <div className={styles.formGroup}>
                <label htmlFor="role">Role *</label>
                <select
                  id="role"
                  required
                  value={createForm.role}
                  onChange={(e) => setCreateForm(prev => ({
                    ...prev,
                    role: e.target.value as 'DISPATCHER' | 'DRIVER'
                  }))}
                >
                  <option value="DRIVER">Driver</option>
                  <option value="DISPATCHER">Dispatcher</option>
                </select>
              </div>
              
              <div className={styles.modalActions}>
                <button
                  type="button"
                  className={styles.cancelButton}
                  onClick={() => {
                    setShowCreateForm(false);
                    setCreateForm({ email: '', password: '', role: 'DRIVER' });
                    setError(null);
                  }}
                >
                  Cancel
                </button>
                <button
                  type="submit"
                  className={styles.submitButton}
                  disabled={submitting}
                >
                  {submitting ? 'Creating...' : 'Create User'}
                </button>
              </div>
            </form>
          </div>
        </div>
      )}

      {/* Reset Password Modal */}
      {resetPasswordUser && (
        <div className={styles.modal}>
          <div className={styles.modalContent}>
            <h2>Reset Password</h2>
            <p>Reset password for: {resetPasswordUser.email}</p>
            <form onSubmit={handleResetPassword}>
              <div className={styles.formGroup}>
                <label htmlFor="newPassword">New Password *</label>
                <input
                  id="newPassword"
                  type="password"
                  required
                  value={resetForm.newPassword}
                  onChange={(e) => setResetForm({ newPassword: e.target.value })}
                />
              </div>
              
              <div className={styles.modalActions}>
                <button
                  type="button"
                  className={styles.cancelButton}
                  onClick={() => {
                    setResetPasswordUser(null);
                    setResetForm({ newPassword: '' });
                    setError(null);
                  }}
                >
                  Cancel
                </button>
                <button
                  type="submit"
                  className={styles.submitButton}
                  disabled={submitting}
                >
                  {submitting ? 'Resetting...' : 'Reset Password'}
                </button>
              </div>
            </form>
          </div>
        </div>
      )}
    </div>
  );
};

export default UsersPage;
```
