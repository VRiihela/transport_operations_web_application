import { z } from 'zod';

export const createJobSchema = z.object({
  // ... existing fields
  scheduledStart: z.string().datetime().optional().nullable(),
  scheduledEnd: z.string().datetime().optional().nullable(),
  schedulingNote: z.string().optional().nullable(),
  // ... rest of existing fields
}).refine(
  (data) => {
    // If both scheduled times are null/undefined, schedulingNote is required
    const hasScheduledStart = data.scheduledStart != null;
    const hasScheduledEnd = data.scheduledEnd != null;
    const hasNote = data.schedulingNote != null && data.schedulingNote.trim().length > 0;
    
    if (!hasScheduledStart && !hasScheduledEnd) {
      return hasNote;
    }
    return true;
  },
  {
    message: "schedulingNote is required when both scheduledStart and scheduledEnd are not provided",
    path: ["schedulingNote"],
  }
);

export const updateJobSchema = createJobSchema.partial();

export type CreateJobInput = z.infer<typeof createJobSchema>;
export type UpdateJobInput = z.infer<typeof updateJobSchema>;

## 5. Finnish Date Formatting Utility

**File: `src/utils/date.ts`**
```typescript
export function formatFinnishDateTime(dateString: string | null): string {
  if (!dateString) return '';
  
  const date = new Date(dateString);
  
  // Finnish locale formatting: DD.MM.YYYY HH:MM
  return date.toLocaleString('fi-FI', {
    day: '2-digit',
    month: '2-digit',
    year: 'numeric',
    hour: '2-digit',
    minute: '2-digit',
    timeZone: 'Europe/Helsinki',
  });
}

export function formatSchedulingInfo(
  scheduledStart: string | null,
  scheduledEnd: string | null,
  schedulingNote: string | null
): string {
  if (scheduledStart || scheduledEnd) {
    const start = scheduledStart ? formatFinnishDateTime(scheduledStart) : '';
    const end = scheduledEnd ? formatFinnishDateTime(scheduledEnd) : '';
    
    if (start && end) {
      return `${start} - ${end}`;
    }
    return start || end;
  }
  
  return schedulingNote || '';
}
```
