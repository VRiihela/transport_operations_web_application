## Task Specification

**Description:** Implement the Dispatcher/Admin jobs view. This replaces the JobsPage placeholder with a fully functional page. Dispatchers can see all jobs, create new jobs, assign drivers, and update job status. Uses the existing backend API at /api/jobs. Keep UI simple and functional — no complex design needed.

**Acceptance Criteria:**
- JobsPage fetches GET /api/jobs on mount and displays jobs in a list/table showing: title, status, assigned driver name (or Unassigned), scheduledAt
- Jobs list shows loading state while fetching and error state if request fails
- Create Job button opens a form (inline or simple modal) with fields: title (required), description (optional), scheduledAt (optional), location (optional)
- Create job form submits POST /api/jobs, adds new job to list on success, shows error on failure

- Assigning a driver calls PATCH /api/jobs/:id with assignedDriverId and updates the list

- Status badges are color coded: DRAFT=gray, ASSIGNED=blue, IN_PROGRESS=yellow, COMPLETED=green
- All API errors show a simple error message to the user — no raw error objects
- TypeScript strict — no any types

```typescript
import React, { useState, useEffect, useCallback } from 'react';
import axios from '../api/axios';
import styles from './JobsPage.module.css';

// TypeScript interfaces
interface Job {
  id: string;
  title: string;
  description?: string;
  status: JobStatus;
  assignedDriverId?: string;
  assignedDriverName?: string;
  scheduledAt?: string;
  location?: string;
  createdAt: string;
  updatedAt: string;
}

interface Driver {
  id: string;
  name: string;
  email: string;
}

interface CreateJobFormData {
  title: string;
  description: string;
  scheduledAt: string;
  location: string;
}

type JobStatus = 'DRAFT' | 'ASSIGNED' | 'IN_PROGRESS' | 'COMPLETED';

const JobsPage: React.FC = () => {
  const [jobs, setJobs] = useState<Job[]>([]);
  const [drivers, setDrivers] = useState<Driver[]>([]);
  const [loading, setLoading] = useState<boolean>(true);
  const [error, setError] = useState<string>('');
  const [showCreateForm, setShowCreateForm] = useState<boolean>(false);
  const [createFormData, setCreateFormData] = useState<CreateJobFormData>({
    title: '',
    description: '',
    scheduledAt: '',
    location: ''
  });
  const [createFormErrors, setCreateFormErrors] = useState<Record<string, string>>({});
  const [submitting, setSubmitting] = useState<boolean>(false);
  const [loadingDrivers, setLoadingDrivers] = useState<boolean>(false);
  const [assigningJobs, setAssigningJobs] = useState<Set<string>>(new Set());
  const [updatingStatus, setUpdatingStatus] = useState<Set<string>>(new Set());

  // Fetch jobs on component mount
  const fetchJobs = useCallback(async () => {
    try {
      setLoading(true);
      setError('');
      const response = await axios.get<Job[]>('/api/jobs');
      setJobs(response.data);
    } catch (err: any) {
      setError('Failed to load jobs. Please try again.');
      console.error('Error fetching jobs:', err);
    } finally {
      setLoading(false);
    }
  }, []);

  // Fetch available drivers
  const fetchDrivers = useCallback(async () => {
    try {
      setLoadingDrivers(true);
      const response = await axios.get<Driver[]>('/api/users?role=DRIVER');
      setDrivers(response.data);
    } catch (err: any) {
      console.error('Error fetching drivers:', err);
    } finally {
      setLoadingDrivers(false);
    }
  }, []);

  useEffect(() => {
    fetchJobs();
  }, [fetchJobs]);

  // Handle create job form submission
  const handleCreateJob = async (e: React.FormEvent) => {
    e.preventDefault();
    
    // Validate form
    const errors: Record<string, string> = {};
    if (!createFormData.title.trim()) {
      errors.title = 'Title is required';
    }
    
    setCreateFormErrors(errors);
    
    if (Object.keys(errors).length > 0) {
      return;
    }

    try {
      setSubmitting(true);
      const payload = {
        title: createFormData.title.trim(),
        description: createFormData.description.trim() || undefined,
        scheduledAt: createFormData.scheduledAt || undefined,
        location: createFormData.location.trim() || undefined
      };
      
      const response = await axios.post<Job>('/api/jobs', payload);
      setJobs(prev => [response.data, ...prev]);
      setShowCreateForm(false);
      setCreateFormData({ title: '', description: '', scheduledAt: '', location: '' });
      setCreateFormErrors({});
    } catch (err: any) {
      const message = err.response?.data?.message || 'Failed to create job. Please try again.';
      setError(message);
    } finally {
      setSubmitting(false);
    }
  };

  // Handle driver assignment
  const handleAssignDriver = async (jobId: string, driverId: string) => {
    try {
      setAssigningJobs(prev => new Set([...prev, jobId]));
      const response = await axios.patch<Job>(`/api/jobs/${jobId}`, {
        assignedDriverId: driverId
      });
      
      setJobs(prev => prev.map(job => 
        job.id === jobId ? response.data : job
      ));
    } catch (err: any) {
      const message = err.response?.data?.message || 'Failed to assign driver. Please try again.';
      setError(message);
    } finally {
      setAssigningJobs(prev => {
        const newSet = new Set(prev);
        newSet.delete(jobId);
        return newSet;
      });
    }
  };

  // Handle status update
  const handleStatusUpdate = async (jobId: string, newStatus: JobStatus) => {
    const originalJobs = [...jobs];
    
    try {
      setUpdatingStatus(prev => new Set([...prev, jobId]));
      
      // Optimistic update
      setJobs(prev => prev.map(job => 
        job.id === jobId ? { ...job, status: newStatus } : job
      ));
      
      const response = await axios.patch<Job>(`/api/jobs/${jobId}`, {
        status: newStatus
      });
      
      // Replace with server response
      setJobs(prev => prev.map(job => 
        job.id === jobId ? response.data : job
      ));
    } catch (err: any) {
      // Rollback optimistic update
      setJobs(originalJobs);
      const message = err.response?.data?.message || 'Failed to update job status. Please try again.';
      setError(message);
    } finally {
      setUpdatingStatus(prev => {
        const newSet = new Set(prev);
        newSet.delete(jobId);
        return newSet;
      });
    }
  };

  // Get allowed next statuses for a job
  const getAllowedStatuses = (currentStatus: JobStatus): JobStatus[] => {
    switch (currentStatus) {
      case 'DRAFT':
        return ['ASSIGNED'];
      case 'ASSIGNED':
        return ['IN_PROGRESS'];
      case 'IN_PROGRESS':
        return ['COMPLETED'];
      case 'COMPLETED':
        return [];
      default:
        return [];
    }
  };

  // Get status badge class
  const getStatusBadgeClass = (status: JobStatus): string => {
    const baseClass = styles.statusBadge;
    switch (status) {
      case 'DRAFT':
        return `${baseClass} ${styles.statusDraft}`;
      case 'ASSIGNED':
        return `${baseClass} ${styles.statusAssigned}`;
      case 'IN_PROGRESS':
        return `${baseClass} ${styles.statusInProgress}`;
      case 'COMPLETED':
        return `${baseClass} ${styles.statusCompleted}`;
      default:
        return baseClass;
    }
  };

  // Format date for display
  const formatDate = (dateString?: string): string => {
    if (!dateString) return 'Not scheduled';
    return new Date(dateString).toLocaleDateString();
  };

  if (loading) {
    return (
      <div className={styles.container}>
        <div className={styles.loading}>Loading jobs...</div>
      </div>
    );
  }

  return (
    <div className={styles.container}>
      <div className={styles.header}>
        <h1>Jobs Management</h1>
        <button 
          className={styles.createButton}
          onClick={() => setShowCreateForm(true)}
        >
          Create Job
        </button>
      </div>

      {error && (
        <div className={styles.error}>
          {error}
          <button 
            className={styles.errorDismiss}
            onClick={() => setError('')}
          >
            ×
          </button>
        </div>
      )}

      {showCreateForm && (
        <div className={styles.modal}>
          <div className={styles.modalContent}>
            <div className={styles.modalHeader}>
              <h2>Create New Job</h2>
              <button 
                className={styles.modalClose}
                onClick={() => {
                  setShowCreateForm(false);
                  setCreateFormData({ title: '', description: '', scheduledAt: '', location: '' });
                  setCreateFormErrors({});
                }}
              >
                ×
              </button>
            </div>
            
            <form onSubmit={handleCreateJob} className={styles.form}>
              <div className={styles.formGroup}>
                <label htmlFor="title">Title *</label>
                <input
                  id="title"
                  type="text"
                  value={createFormData.title}
                  onChange={(e) => setCreateFormData(prev => ({ ...prev, title: e.target.value }))}
                  className={createFormErrors.title ? styles.inputError : ''}
                  maxLength={200}
                />
                {createFormErrors.title && (
                  <span className={styles.fieldError}>{createFormErrors.title}</span>
                )}
              </div>

              <div className={styles.formGroup}>
                <label htmlFor="description">Description</label>
                <textarea
                  id="description"
                  value={createFormData.description}
                  onChange={(e) => setCreateFormData(prev => ({ ...prev, description: e.target.value }))}
                  rows={3}
                  maxLength={1000}
                />
              </div>

              <div className={styles.formGroup}>
                <label htmlFor="scheduledAt">Scheduled Date</label>
                <input
                  id="scheduledAt"
                  type="datetime-local"
                  value={createFormData.scheduledAt}
                  onChange={(e) => setCreateFormData(prev => ({ ...prev, scheduledAt: e.target.value }))}
                />
              </div>

              <div className={styles.formGroup}>
                <label htmlFor="location">Location</label>
                <input
                  id="location"
                  type="text"
                  value={createFormData.location}
                  onChange={(e) => setCreateFormData(prev => ({ ...prev, location: e.target.value }))}
                  maxLength={500}
                />
              </div>

              <div className={styles.formActions}>
                <button type="submit" disabled={submitting}>
                  {submitting ? 'Creating...' : 'Create Job'}
                </button>
                <button 
                  type="button" 
                  onClick={() => setShowCreateForm(false)}
                  className={styles.cancelButton}
                >
                  Cancel
                </button>
              </div>
            </form>
          </div>
        </div>
      )}

      {jobs.length === 0 ? (
        <div className={styles.emptyState}>
          <p>No jobs found. Create your first job to get started.</p>
        </div>
      ) : (
        <div className={styles.tableContainer}>
          <table className={styles.jobsTable}>
            <thead>
              <tr>
                <th>Title</th>
                <th>Status</th>
                <th>Assigned Driver</th>
                <th>Scheduled</th>
                <th>Actions</th>
              </tr>
            </thead>
            <tbody>
              {jobs.map((job) => (
                <tr key={job.id}>
                  <td>
                    <div className={styles.jobTitle}>{job.title}</div>
                    {job.description && (
                      <div className={styles.jobDescription}>{job.description}</div>
                    )}
                  </td>
                  <td>
                    <span className={getStatusBadgeClass(job.status)}>
                      {job.status.replace('_', ' ')}
                    </span>
                  </td>
                  <td>
                    {job.assignedDriverName || 'Unassigned'}
                  </td>
                  <td>{formatDate(job.scheduledAt)}</td>
                  <td className={styles.actions}>
                    {job.status === 'DRAFT' && (
                      <div className={styles.assignDriverContainer}>
                        <button
                          className={styles.assignButton}
                          onClick={() => fetchDrivers()}
                          disabled={assigningJobs.has(job.id)}
                        >
                          {assigningJobs.has(job.id) ? 'Assigning...' : 'Assign Driver'}
                        </button>
                        {drivers.length > 0 && (
                          <select
                            className={styles.driverSelect}
                            onChange={(e) => {
                              if (e.target.value) {
                                handleAssignDriver(job.id, e.target.value);
                              }
                            }}
                            value=""
                            disabled={loadingDrivers || assigningJobs.has(job.id)}
                          >
                            <option value="">Select driver...</option>
                            {drivers.map((driver) => (
                              <option key={driver.id} value={driver.id}>
                                {driver.name}
                              </option>
                            ))}
                          </select>
                        )}
                      </div>
                    )}
                    
                    {getAllowedStatuses(job.status).length > 0 && (
                      <div className={styles.statusActions}>
                        {getAllowedStatuses(job.status).map((status) => (
                          <button
                            key={status}
                            className={styles.statusButton}
                            onClick={() => handleStatusUpdate(job.id, status)}
                            disabled={updatingStatus.has(job.id)}
                          >
                            {updatingStatus.has(job.id) ? 'Updating...' : `Mark ${status.replace('_', ' ')}`}
                          </button>
                        ))}
                      </div>
                    )}
                  </td>
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      )}
    </div>
  );
};

export default JobsPage;
```
