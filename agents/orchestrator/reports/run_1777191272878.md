### src/components/JobCard.tsx (new file)

```typescript
import React from 'react';
import { Job } from '../types';
import styles from '../styles/JobCard.module.css';

interface JobCardProps {
  job: Job;
}

const JobCard: React.FC<JobCardProps> = ({ job }) => {
  const getPriorityClass = (priority: Job['priority']): string => {
    switch (priority) {
      case 'urgent':
        return styles.priorityUrgent;
      case 'high':
        return styles.priorityHigh;
      case 'medium':
        return styles.priorityMedium;
      case 'low':
        return styles.priorityLow;
      default:
        return styles.priorityMedium;
    }
  };

  const formatDuration = (minutes: number): string => {
    if (minutes < 60) {
      return `${minutes}m`;
    }
    const hours = Math.floor(minutes / 60);
    const remainingMinutes = minutes % 60;
    return remainingMinutes > 0 ? `${hours}h ${remainingMinutes}m` : `${hours}h`;
  };

  return (
    <div className={`${styles.jobCard} ${getPriorityClass(job.priority)}`}>
      <div className={styles.header}>
        <h3 className={styles.title}>{job.title}</h3>
        <span className={styles.priority}>{job.priority}</span>
      </div>
      <p className={styles.description}>{job.description}</p>
      <div className={styles.details}>
        <div className={styles.location}>
          <span className={styles.locationIcon}>📍</span>
          <span className={styles.locationText}>{job.location.address}</span>
        </div>
        <div className={styles.duration}>
          <span className={styles.durationIcon}>⏱️</span>
          <span className={styles.durationText}>{formatDuration(job.estimatedDuration)}</span>
        </div>
      </div>
      <div className={styles.footer}>
        <span className={styles.status}>{job.status.replace('_', ' ')}</span>
        <span className={styles.createdAt}>
          {job.createdAt.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' })}
        </span>
      </div>
    </div>
  );
};

export default JobCard;
```

```typescript
import React from 'react';
import { Job } from '../types';

interface JobPoolProps {
  jobs: Job[];
}

const JobPool: React.FC<JobPoolProps> = ({ jobs }) => {
  const unassignedJobs = jobs.filter(job => job.status === 'unassigned');

  return (
    <div className={styles.jobPool}>
      <div className={styles.header}>
        <h2 className={styles.title}>Unassigned Jobs</h2>
        <span className={styles.count}>({unassignedJobs.length})</span>
      </div>
      <div className={styles.jobList}>
        {unassignedJobs.length === 0 ? (
          <div className={styles.emptyState}>
            <span className={styles.emptyIcon}>📋</span>

          </div>
        ) : (
          unassignedJobs.map(job => (
            <JobCard key={job.id} job={job} />
          ))

    </div>
  );
};
```
