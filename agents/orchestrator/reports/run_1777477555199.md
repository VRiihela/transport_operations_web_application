import { 
  startOfWeek, 
  endOfWeek, 
  addWeeks, 
  subWeeks, 
  format, 
  parseISO, 
  isValid,
  startOfDay,
  addDays,
  isSameDay
} from 'date-fns';
import { fi } from 'date-fns/locale';

export const formatDayHeader = (date: Date): string => {
  if (!isValid(date)) return '';
  return format(date, 'E d.M.', { locale: fi });
};

export const getWeekStart = (date: Date): Date => {
  return startOfWeek(date, { weekStartsOn: 1 }); // Monday start
};

export const getWeekEnd = (date: Date): Date => {
  return endOfWeek(date, { weekStartsOn: 1 });
};

export const getWeekDays = (weekStart: Date): Date[] => {
  const days: Date[] = [];
  for (let i = 0; i < 7; i++) {
    days.push(addDays(weekStart, i));
  }
  return days;
};

export const navigateWeek = (currentWeek: Date, direction: 'prev' | 'next'): Date => {
  return direction === 'prev' ? subWeeks(currentWeek, 1) : addWeeks(currentWeek, 1);
};

export const parseScheduledStart = (scheduledStart: Date | string | undefined): Date | null => {
  if (!scheduledStart) return null;
  
  if (scheduledStart instanceof Date) {
    return isValid(scheduledStart) ? scheduledStart : null;
  }
  
  if (typeof scheduledStart === 'string') {
    const parsed = parseISO(scheduledStart);
    return isValid(parsed) ? parsed : null;
  }
  
  return null;
};

export const sortJobsByTime = (jobs: Job[]): Job[] => {
  return [...jobs].sort((a, b) => {
    const timeA = parseScheduledStart(a.scheduledStart);
    const timeB = parseScheduledStart(b.scheduledStart);
    
    if (!timeA && !timeB) return 0;
    if (!timeA) return 1;
    if (!timeB) return -1;
    
    return timeA.getTime() - timeB.getTime();
  });
};

export const isJobOnDate = (job: Job, targetDate: Date): boolean => {
  const jobDate = parseScheduledStart(job.scheduledStart);
  if (!jobDate) return false;
  return isSameDay(jobDate, targetDate);
};

export const validateDateInput = (dateString: string): Date | null => {
  if (!dateString || typeof dateString !== 'string') return null;
  
  const parsed = parseISO(dateString);
  return isValid(parsed) ? parsed : null;
};

#### `src/hooks/useWeekNavigation.ts`

import { useState, useCallback, useMemo } from 'react';
import { getWeekStart, navigateWeek, validateDateInput, getWeekDays } from '../utils/dateUtils';

export const useWeekNavigation = (initialDate?: Date) => {
  const [currentWeek, setCurrentWeek] = useState<Date>(
    getWeekStart(initialDate || new Date())
  );

  const weekDays = useMemo(() => getWeekDays(currentWeek), [currentWeek]);

  const goToPrevWeek = useCallback(() => {
    setCurrentWeek(prev => navigateWeek(prev, 'prev'));
  }, []);

  const goToNextWeek = useCallback(() => {
    setCurrentWeek(prev => navigateWeek(prev, 'next'));
  }, []);

  const goToWeek = useCallback((date: Date | string) => {
    let targetDate: Date;
    
    if (typeof date === 'string') {
      const parsed = validateDateInput(date);
      if (!parsed) return false;
      targetDate = parsed;
    } else {
      targetDate = date;
    }
    
    setCurrentWeek(getWeekStart(targetDate));
    return true;
  }, []);

  return {
    currentWeek,
    weekDays,
    goToPrevWeek,
    goToNextWeek,
    goToWeek
  };
};

import React, { useState, useCallback } from 'react';
import { WeekNavigationProps } from '../../types/job.types';
import { format } from 'date-fns';
import { fi } from 'date-fns/locale';

export const WeekNavigation: React.FC<WeekNavigationProps> = ({
  currentWeek,
  onWeekChange,
  onPrevWeek,
  onNextWeek
}) => {
  const [dateInput, setDateInput] = useState('');
  const [inputError, setInputError] = useState('');

  const handleDateInputChange = useCallback((e: React.ChangeEvent<HTMLInputElement>) => {
    const value = e.target.value;
    setDateInput(value);
    setInputError('');
  }, []);

  const handleDateSubmit = useCallback((e: React.FormEvent) => {
    e.preventDefault();
    
    if (!dateInput.trim()) {
      setInputError('Syötä päivämäärä');
      return;
    }

    try {
      const targetDate = new Date(dateInput);
      if (isNaN(targetDate.getTime())) {
        setInputError('Virheellinen päivämäärä');
        return;
      }

      onWeekChange(targetDate);
      setDateInput('');
      setInputError('');
    } catch (error) {
      setInputError('Virheellinen päivämäärä');
    }
  }, [dateInput, onWeekChange]);

  const currentWeekLabel = format(currentWeek, "'Viikko' I/yyyy", { locale: fi });

  return (
    <div className="week-navigation">
      <div className="nav-controls">
        <button 
          type="button"
          className="nav-button prev"
          onClick={onPrevWeek}
          aria-label="Edellinen viikko"
        >
          ←
        </button>
        
        <div className="current-week-display">
          {currentWeekLabel}
        </div>
        
        <button 
          type="button"
          className="nav-button next"
          onClick={onNextWeek}
          aria-label="Seuraava viikko"
        >
          →
        </button>
      </div>
      
      <form className="date-picker" onSubmit={handleDateSubmit}>
        <input
          type="date"
          value={dateInput}
          onChange={handleDateInputChange}
          placeholder="Siirry päivämäärään..."
          aria-label="Valitse päivämäärä"
        />
        <button type="submit" className="date-go-button">
          Siirry
        </button>
        {inputError && (
          <div className="input-error" role="alert">
            {inputError}
          </div>
        )}
      </form>
    </div>
  );
};
