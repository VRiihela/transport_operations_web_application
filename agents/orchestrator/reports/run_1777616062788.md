import { Job } from '../models/Job';
import { Driver } from '../models/Driver';
import { Team } from '../models/Team';

export class JobService {
  async getDriverJobsForDate(driverId: string, date: string): Promise<JobWithAssignment[]> {
    if (!driverId || !date) {
      throw new Error('Driver ID and date are required');
    }

    // Validate date format (YYYY-MM-DD)
    const dateRegex = /^\d{4}-\d{2}-\d{2}$/;
    if (!dateRegex.test(date)) {
      throw new Error('Invalid date format. Expected YYYY-MM-DD');
    }

    const targetDate = new Date(date);
    if (isNaN(targetDate.getTime())) {
      throw new Error('Invalid date value');
    }

    // Get directly assigned jobs
    const directJobs = await Job.findAll({
      where: {
        assignedDriverId: driverId,
        scheduledDate: date
      },
      include: [
        {
          model: Driver,
          as: 'assignedDriver',
          attributes: ['id', 'name']
        }
      ]
    });

    // Get team-assigned jobs for driver's teams
    const teamJobs = await Job.findAll({
      include: [
        {
          model: Team,
          as: 'assignedTeam',
          attributes: ['id', 'name'],
          include: [
            {
              model: Driver,
              as: 'members',
              where: { id: driverId },
              attributes: []
            }
          ]
        }
      ],
      where: {
        assignedTeamId: { [Op.ne]: null },
        scheduledDate: date,
        assignedDriverId: null // Only team-assigned, not direct
      }
    });

    // Combine and format results
    const jobsWithAssignment: JobWithAssignment[] = [
      ...directJobs.map(job => ({
        ...job.toJSON(),
        assignmentType: 'direct' as const,
        teamName: null
      })),
      ...teamJobs.map(job => ({
        ...job.toJSON(),
        assignmentType: 'team' as const,
        teamName: job.assignedTeam?.name || null
      }))
    ];

    return jobsWithAssignment.sort((a, b) => 
      new Date(a.scheduledTime).getTime() - new Date(b.scheduledTime).getTime()
    );
  }

  async updateJobStatus(jobId: string, driverId: string, status: JobStatus, notes?: string): Promise<Job> {
    if (!jobId || !driverId || !status) {
      throw new Error('Job ID, driver ID, and status are required');
    }

    // Validate status is a valid enum value
    const validStatuses = ['pending', 'in_progress', 'completed', 'cancelled'];
    if (!validStatuses.includes(status)) {
      throw new Error('Invalid job status');
    }

    // Sanitize notes input
    const sanitizedNotes = notes ? notes.trim().substring(0, 500) : undefined;

    const job = await Job.findByPk(jobId, {
      include: [
        {
          model: Team,
          as: 'assignedTeam',
          include: [
            {
              model: Driver,
              as: 'members',
              attributes: ['id']
            }
          ]
        }
      ]
    });

    if (!job) {
      throw new Error('Job not found');
    }

    // Check authorization: direct assignment OR team membership
    const isDirectlyAssigned = job.assignedDriverId === driverId;
    const isTeamMember = job.assignedTeam?.members?.some(member => member.id === driverId) || false;

    if (!isDirectlyAssigned && !isTeamMember) {
      throw new Error('Driver not authorized to update this job');
    }

    await job.update({
      status,
      notes: sanitizedNotes,
      updatedAt: new Date()
    });

    return job;
  }
}

import { Router } from 'express';
import { JobService } from '../../services/JobService';
import { authenticateDriver } from '../../middleware/auth';
import { UpdateJobStatusRequest } from '../../types/Job';

const router = Router();
const jobService = new JobService();

// Get driver jobs for specific date
router.get('/:date', authenticateDriver, async (req, res) => {
  try {
    const driverId = req.driver.id; // From auth middleware
    const { date } = req.params;

    if (!date) {
      return res.status(400).json({
        error: 'Date parameter is required'
      });
    }

    const jobs = await jobService.getDriverJobsForDate(driverId, date);

    res.json({
      jobs,
      totalCount: jobs.length,
      date
    });
  } catch (error) {
    console.error('Error fetching driver jobs:', error);
    
    if (error.message.includes('Invalid date')) {
      return res.status(400).json({
        error: error.message
      });
    }

    res.status(500).json({
      error: 'Failed to fetch jobs'
    });
  }
});

// Update job status
router.patch('/:jobId/status', authenticateDriver, async (req, res) => {
  try {
    const driverId = req.driver.id;
    const { jobId } = req.params;
    const { status, notes }: UpdateJobStatusRequest = req.body;

    if (!jobId) {
      return res.status(400).json({
        error: 'Job ID is required'
      });
    }

    const updatedJob = await jobService.updateJobStatus(jobId, driverId, status, notes);

    res.json({
      job: updatedJob,
      message: 'Job status updated successfully'
    });
  } catch (error) {
    console.error('Error updating job status:', error);

    if (error.message.includes('not authorized')) {
      return res.status(403).json({
        error: 'Not authorized to update this job'
      });
    }

    if (error.message.includes('not found')) {
      return res.status(404).json({
        error: 'Job not found'
      });
    }

    if (error.message.includes('Invalid')) {
      return res.status(400).json({
        error: error.message
      });
    }

    res.status(500).json({
      error: 'Failed to update job status'
    });
  }
});

export default router;
