```typescript
import { Job } from '../models/Job';
import { ValidationError, NotFoundError } from '../utils/errors';

export class JobService {
  async updateJobOrder(jobId: number, sortOrder: number, userId: number): Promise<Job> {
    // Validate sortOrder
    if (!Number.isInteger(sortOrder) || sortOrder < 1 || sortOrder > 99999) {
      throw new ValidationError('sortOrder must be an integer between 1 and 99999');
    }

    // Find job and verify ownership/permissions
    const job = await Job.findByPk(jobId);
    if (!job) {
      throw new NotFoundError('Job not found');
    }

    // Update sortOrder
    job.sortOrder = sortOrder;
    await job.save();

    return job;
  }

  async reorderJobsInColumn(driverIdjobs: Array<{id: number; sortOrder: number}>, driverId?: number): Promise<void> {
    // Validate all jobs belong to the same driver
    const jobIds = driverIdjobs.map(j => j.id);
    const jobs = await Job.findAll({
      where: { 
        id: jobIds,
        assignedDriverId: driverId || null
      }
    });

    if (jobs.length !== jobIds.length) {
      throw new ValidationError('Some jobs do not belong to the specified driver');
    }

    // Update in transaction
    await Job.sequelize?.transaction(async (transaction) => {
      for (const { id, sortOrder } of driverIdjobs) {
        if (!Number.isInteger(sortOrder) || sortOrder < 1) {
          throw new ValidationError(`Invalid sortOrder for job ${id}`);
        }
        
        await Job.update(
          { sortOrder },
          { where: { id }, transaction }
        );
      }
    });
  }

  async getJobsByDriver(driverId?: number): Promise<Job[]> {
    return Job.findAll({
      where: { assignedDriverId: driverId || null },
      order: [['sortOrder', 'ASC'], ['createdAt', 'ASC']]
    });
  }
}
```

```typescript
import { Request, Response, NextFunction } from 'express';
import { JobService } from '../services/jobService';
import { ValidationError, NotFoundError } from '../utils/errors';
import { body, param, validationResult } from 'express-validator';

export class JobController {
  private jobService: JobService;

  constructor() {
    this.jobService = new JobService();
  }

  updateJobOrder = async (req: Request, res: Response, next: NextFunction): Promise<void> => {
    try {
      const errors = validationResult(req);
      if (!errors.isEmpty()) {
        res.status(400).json({
          success: false,
          message: 'Validation failed',
          errors: errors.array()
        });
        return;
      }

      const jobId = parseInt(req.params.id, 10);
      const { sortOrder } = req.body;
      const userId = req.user?.id; // Assuming auth middleware sets user

      if (!userId) {
        res.status(401).json({ success: false, message: 'Unauthorized' });
        return;
      }

      const job = await this.jobService.updateJobOrder(jobId, sortOrder, userId);

      res.json({
        success: true,
        data: job,
        message: 'Job order updated successfully'
      });
    } catch (error) {
      next(error);
    }
  };

  reorderJobs = async (req: Request, res: Response, next: NextFunction): Promise<void> => {
    try {
      const errors = validationResult(req);
      if (!errors.isEmpty()) {
        res.status(400).json({
          success: false,
          message: 'Validation failed',
          errors: errors.array()
        });
        return;
      }

      const { jobs, driverId } = req.body;

      await this.jobService.reorderJobsInColumn(jobs, driverId);

      res.json({
        success: true,
        message: 'Jobs reordered successfully'
      });
    } catch (error) {
      next(error);
    }
  };
}
```

```typescript
import { Job, JobReorderRequest, ApiResponse } from '../types/Job';

const API_BASE = '/api/jobs';

export const jobApi = {
  async updateJobOrder(jobId: number, sortOrder: number): Promise<Job> {
    const response = await fetch(`${API_BASE}/${jobId}`, {
      method: 'PATCH',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${localStorage.getItem('token')}`,
      },
      body: JSON.stringify({ sortOrder }),
    });

    if (!response.ok) {
      const error: ApiResponse<never> = await response.json();
      throw new Error(error.message || 'Failed to update job order');
    }

    const result: ApiResponse<Job> = await response.json();
    if (!result.success || !result.data) {
      throw new Error(result.message || 'Failed to update job order');
    }

    return result.data;
  },

  async reorderJobs(request: JobReorderRequest): Promise<void> {
    const response = await fetch(`${API_BASE}/reorder`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${localStorage.getItem('token')}`,
      },
      body: JSON.stringify(request),
    });

    if (!response.ok) {
      const error: ApiResponse<never> = await response.json();
      throw new Error(error.message || 'Failed to reorder jobs');
    }

    const result: ApiResponse<never> = await response.json();
    if (!result.success) {
      throw new Error(result.message || 'Failed to reorder jobs');
    }
  },
```

```typescript
import { useState, useCallback } from 'react';
import { Job } from '../types/Job';
import { jobApi } from '../services/jobApi';

export interface UseJobReorderReturn {
  isReordering: boolean;
  reorderError: string | null;
  reorderJobs: (jobs: Job[], driverId?: number) => Promise<void>;
  updateSingleJobOrder: (jobId: number, newSortOrder: number) => Promise<void>;
}

export const useJobReorder = (onSuccess?: () => void): UseJobReorderReturn => {
  const [isReordering, setIsReordering] = useState(false);
  const [reorderError, setReorderError] = useState<string | null>(null);

  const reorderJobs = useCallback(async (jobs: Job[], driverId?: number) => {
    setIsReordering(true);
    setReorderError(null);

    try {
      const reorderData = {
        jobs: jobs.map((job, index) => ({
          id: job.id,
          sortOrder: (index + 1) * 100, // Space between items
        })),
        driverId,
      };

      await jobApi.reorderJobs(reorderData);
      onSuccess?.();
    } catch (error) {
      const message = error instanceof Error ? error.message : 'Failed to reorder jobs';
      setReorderError(message);
      throw error;
    } finally {
      setIsReordering(false);
    }
  }, [onSuccess]);

  const updateSingleJobOrder = useCallback(async (jobId: number, newSortOrder: number) => {
    setIsReordering(true);
    setReorderError(null);

    try {
      await jobApi.updateJobOrder(jobId, newSortOrder);
      onSuccess?.();
    } catch (error) {
      const message = error instanceof Error ? error.message : 'Failed to update job order';
      setReorderError(message);
      throw error;
    } finally {
      setIsReordering(false);
    }
  }, [onSuccess]);

  return {
    isReordering,
    reorderError,
    reorderJobs,
    updateSingleJobOrder,
  };
};
```
