import rateLimit from 'express-rate-limit';
import { Request, Response } from 'express';

const createRateLimitHandler = (windowMs: number, max: number, message: string) => {
  return rateLimit({
    windowMs,
    max,
    standardHeaders: true,
    legacyHeaders: false,
    handler: (req: Request, res: Response) => {
      const retryAfter = Math.ceil(windowMs / 1000);
      res.set('Retry-After', retryAfter.toString());
      res.status(429).json({
        success: false,
        error: {
          code: 'RATE_LIMIT_EXCEEDED',
          message,
          retryAfter
        }
      });
    },
    keyGenerator: (req: Request) => {
      // Extract IP with proper proxy handling
      const forwarded = req.get('X-Forwarded-For');
      if (forwarded) {
        // Take first IP from comma-separated list and validate
        const firstIp = forwarded.split(',')[0].trim();
        // Basic IPv4/IPv6 validation
        if (/^(?:[0-9]{1,3}\.){3}[0-9]{1,3}$/.test(firstIp) || 
            /^(?:[0-9a-fA-F]{1,4}:){7}[0-9a-fA-F]{1,4}$/.test(firstIp)) {
          return firstIp;
        }
      }
      return req.ip || 'unknown';
    }
  });
};

export const loginRateLimit = createRateLimitHandler(
  15 * 60 * 1000, // 15 minutes
  10, // max 10 requests
  'Too many login attempts. Please try again in 15 minutes.'
);

export const refreshRateLimit = createRateLimitHandler(
  15 * 60 * 1000, // 15 minutes
  30, // max 30 requests
  'Too many token refresh requests. Please try again in 15 minutes.'
);

import { PrismaClient, AuditEvent } from '@prisma/client';
import { CreateAuditLogParams, AuditMetadata } from '../types/audit.types';

const prisma = new PrismaClient();

const METADATA_SIZE_LIMIT = 1024; // 1KB

export class AuditService {
  private static extractIpAddress(req: any): string {
    // Extract IP with X-Forwarded-For handling
    const forwarded = req.get?.('X-Forwarded-For');
    if (forwarded) {
      const firstIp = forwarded.split(',')[0].trim();
      // Basic validation - reject obviously malicious values
      if (firstIp.length > 45 || /<|>|script/i.test(firstIp)) {
        return req.ip || 'unknown';
      }
      return firstIp;
    }
    return req.ip || 'unknown';
  }

  private static sanitizeUserAgent(userAgent?: string): string | undefined {
    if (!userAgent) return undefined;
    
    // Limit length and remove potentially malicious content
    const sanitized = userAgent
      .slice(0, 500)
      .replace(/<script[^>]*>.*?<\/script>/gi, '')
      .replace(/[<>]/g, '');
    
    return sanitized || undefined;
  }

  private static validateMetadata(metadata?: AuditMetadata): AuditMetadata | undefined {
    if (!metadata) return undefined;

    const serialized = JSON.stringify(metadata);
    if (serialized.length > METADATA_SIZE_LIMIT) {
      throw new Error(`Audit metadata exceeds ${METADATA_SIZE_LIMIT} bytes limit`);
    }

    // Remove any potentially sensitive fields
    const cleaned = { ...metadata };
    delete cleaned.password;
    delete cleaned.token;
    delete cleaned.refreshToken;

    return cleaned;
  }

  static async logEvent(params: CreateAuditLogParams): Promise<void> {
    try {
      const { event, userId, ip, userAgent, metadata } = params;

      // Validate event is a proper enum value
      if (!Object.values(AuditEvent).includes(event as AuditEvent)) {
        throw new Error(`Invalid audit event: ${event}`);
      }

      const sanitizedMetadata = this.validateMetadata(metadata);
      const sanitizedUserAgent = this.sanitizeUserAgent(userAgent);

      await prisma.auditLog.create({
        data: {
          event: event as AuditEvent,
          userId: userId || null,
          ip: ip.slice(0, 45), // Ensure IP fits in database field
          userAgent: sanitizedUserAgent,
          metadata: sanitizedMetadata || undefined
        }
      });

      // Also log to stdout for immediate visibility
      console.log(`[AUDIT] ${event}: ${ip} ${userId ? `(user: ${userId})` : ''}`, 
        sanitizedMetadata);

    } catch (error) {
      // Don't let audit logging failures break the main flow
      console.error('[AUDIT ERROR]', error);
    }
  }

  static async logFromRequest(
    req: any, 
    event: CreateAuditLogParams['event'],
    userId?: string,
    metadata?: AuditMetadata
  ): Promise<void> {
    const ip = this.extractIpAddress(req);
    const userAgent = req.get?.('User-Agent');

    await this.logEvent({
      event,
      userId,
      ip,
      userAgent,
      metadata
    });
  }
}

import { PrismaClient } from '@prisma/client';

const prisma = new PrismaClient();

export async function cleanupExpiredTokens(): Promise<{ deletedCount: number }> {
  try {
    console.log('[CLEANUP] Starting expired token cleanup...');
    
    const result = await prisma.refreshToken.deleteMany({
      where: {
        OR: [
          {
            // Delete expired tokens
            expiresAt: {
              lt: new Date()
            }
          },
          {
            // Delete revoked tokens
            revokedAt: {
              not: null
            }
          }
        ]
      }
    });

    console.log(`[CLEANUP] Deleted ${result.count} expired/revoked tokens`);
    
    return { deletedCount: result.count };
    
  } catch (error) {
    console.error('[CLEANUP ERROR]', error);
    throw error;
  } finally {
    await prisma.$disconnect();
  }
}

// Allow direct execution
if (require.main === module) {
  cleanupExpiredTokens()
    .then(({ deletedCount }) => {
      console.log(`Cleanup completed. Deleted ${deletedCount} tokens.`);
      process.exit(0);
    })
    .catch((error) => {
      console.error('Cleanup failed:', error);
      process.exit(1);
    });
}
