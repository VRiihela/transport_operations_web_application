import { z } from 'zod'

export const createJobSchema = z.object({
  // ... existing fields ...
  scheduledStart: z.string().datetime().optional().nullable(),
  scheduledEnd: z.string().datetime().optional().nullable(),
  schedulingNote: z.string().optional().nullable()
}).refine(
  (data) => {
    // If both scheduledStart and scheduledEnd are null/undefined, schedulingNote is required
    const hasNoScheduledTimes = !data.scheduledStart && !data.scheduledEnd
    if (hasNoScheduledTimes) {
      return data.schedulingNote && data.schedulingNote.trim().length > 0
    }
    return true
  },
  {
    message: "schedulingNote is required when no scheduled times are provided",
    path: ["schedulingNote"]
  }
)

export const updateJobSchema = createJobSchema.partial()

export type CreateJobInput = z.infer<typeof createJobSchema>
export type UpdateJobInput = z.infer<typeof updateJobSchema>

      }

### 5. Date Formatting Utility

export const formatFinnishDateTime = (dateString: string | null): string => {
  if (!dateString) return ''
  
  try {
    const date = new Date(dateString)
    return new Intl.DateTimeFormat('fi-FI', {
      day: 'numeric',
      month: 'numeric',
      year: 'numeric',
      hour: '2-digit',
      minute: '2-digit',
      hour12: false
    }).format(date)
  } catch (error) {
    console.error('Error formatting date:', error)
    return ''
  }
}

export const formatSchedulingInfo = (
  scheduledStart: string | null,
  scheduledEnd: string | null,
  schedulingNote: string | null
): string => {
  if (scheduledStart || scheduledEnd) {
    const startStr = scheduledStart ? formatFinnishDateTime(scheduledStart) : '-'
    const endStr = scheduledEnd ? formatFinnishDateTime(scheduledEnd) : '-'
    return `${startStr} - ${endStr}`
  }
  
  return schedulingNote || 'No scheduling info'
}
