import { Request, Response } from 'express';
import { Job } from '../models/Job';
import { AppError } from '../utils/AppError';

interface UpdateJobRequest extends Request {
  params: { id: string };
  body: {
    title?: string;
    description?: string;
    assignedDriverId?: string;
    scheduledStart?: string;
    scheduledEnd?: string;
    schedulingNote?: string;
    pickupAddress?: {
      street: string;
      city: string;
      state: string;
      zipCode: string;
    };
    deliveryAddress?: {
      street: string;
      city: string;
      state: string;
      zipCode: string;
    };
  };
}

export const jobController = {
  // Existing methods...

  updateJob: async (req: UpdateJobRequest, res: Response): Promise<void> => {
    try {
      const { id } = req.params;
      const updates = req.body;

      // Check if job exists
      const existingJob = await Job.findById(id);
      if (!existingJob) {
        throw new AppError('Job not found', 404);
      }

      // Validate scheduled dates if both are provided
      if (updates.scheduledStart && updates.scheduledEnd) {
        const startDate = new Date(updates.scheduledStart);
        const endDate = new Date(updates.scheduledEnd);
        
        if (startDate >= endDate) {
          throw new AppError('Scheduled start must be before scheduled end', 400);
        }
      }

      // Update only provided fields
      const updatedJob = await Job.findByIdAndUpdate(
        id,
        { 
          ...updates,
          updatedAt: new Date()
        },
        { 
          new: true, 
          runValidators: true 
        }
      ).populate('assignedDriverId', 'name email');

      res.status(200).json({
        success: true,
        data: updatedJob
      });
    } catch (error) {
      if (error instanceof AppError) {
        res.status(error.statusCode).json({
          success: false,
          message: error.message
        });
      } else {
        res.status(500).json({
          success: false,
          message: 'Internal server error'
        });
      }
    }
  }
};

import { Request, Response, NextFunction } from 'express';
import Joi from 'joi';
import { AppError } from '../utils/AppError';

const addressSchema = Joi.object({
  street: Joi.string().trim().min(1).max(255).required(),
  city: Joi.string().trim().min(1).max(100).required(),
  state: Joi.string().trim().length(2).uppercase().required(),
  zipCode: Joi.string().trim().pattern(/^\d{5}(-\d{4})?$/).required()
});

const updateJobSchema = Joi.object({
  title: Joi.string().trim().min(1).max(255).optional(),
  description: Joi.string().trim().max(1000).allow('').optional(),
  assignedDriverId: Joi.string().pattern(/^[0-9a-fA-F]{24}$/).optional().allow(null),
  scheduledStart: Joi.date().iso().optional(),
  scheduledEnd: Joi.date().iso().greater(Joi.ref('scheduledStart')).optional(),
  schedulingNote: Joi.string().trim().max(500).allow('').optional(),
  pickupAddress: addressSchema.optional(),
  deliveryAddress: addressSchema.optional()
}).min(1); // At least one field must be provided

export const validateJobUpdate = (req: Request, res: Response, next: NextFunction): void => {
  const { error } = updateJobSchema.validate(req.body, {
    abortEarly: false,
    stripUnknown: true
  });

  if (error) {
    const errorMessage = error.details.map(detail => detail.message).join(', ');
    throw new AppError(`Validation error: ${errorMessage}`, 400);
  }

  // Sanitize string fields
  if (req.body.title) {
    req.body.title = req.body.title.trim();
  }
  if (req.body.description) {
    req.body.description = req.body.description.trim();
  }
  if (req.body.schedulingNote) {
    req.body.schedulingNote = req.body.schedulingNote.trim();
  }

  next();
};
