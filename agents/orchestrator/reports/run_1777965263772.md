import { Request, Response } from 'express';
import { JobService } from '../services/job.service';
import { JobType } from '../types/job.types';

export class JobController {
  private jobService = new JobService();

  async createJob(req: Request, res: Response): Promise<void> {
    try {
      // Validation is already handled by Zod middleware
      const jobData = req.body;
      
      // Server-side validation for customerId if provided
      if (jobData.customerId) {
        const customerExists = await prisma.customer.findUnique({
          where: { id: jobData.customerId }
        });
        if (!customerExists) {
          res.status(400).json({ 
            error: 'Invalid customerId: Customer not found' 
          });
          return;
        }
      }

      const job = await this.jobService.createJob(jobData);
      res.status(201).json(job);
    } catch (error) {
      console.error('Error creating job:', error);
      res.status(500).json({ error: 'Internal server error' });
    }
  }

  async updateJob(req: Request, res: Response): Promise<void> {
    try {
      const { id } = req.params;
      const updateData = req.body;

      // Server-side validation for customerId if provided
      if (updateData.customerId) {
        const customerExists = await prisma.customer.findUnique({
          where: { id: updateData.customerId }
        });
        if (!customerExists) {
          res.status(400).json({ 
            error: 'Invalid customerId: Customer not found' 
          });
          return;
        }
      }

      const job = await this.jobService.updateJob(id, updateData);
      res.status(200).json(job);
    } catch (error) {
      console.error('Error updating job:', error);
      res.status(500).json({ error: 'Internal server error' });
    }
  }
