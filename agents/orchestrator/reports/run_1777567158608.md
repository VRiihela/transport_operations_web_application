import { Request, Response } from 'express';
import { PrismaClient } from '@prisma/client';
import { CreateTeamRequest, UpdateTeamRequest, TeamResponse } from '../types/team';

const prisma = new PrismaClient();

export class TeamController {
  static async createTeam(req: Request, res: Response): Promise<void> {
    try {
      const { name, date, driverIds }: CreateTeamRequest = req.body;
      const createdById = req.user?.id;

      // Validate required fields
      if (!name?.trim() || !date || !Array.isArray(driverIds) || driverIds.length === 0) {
        res.status(400).json({ 
          success: false, 
          error: 'Name, date, and at least one driver are required' 
        });
        return;
      }

      // Validate date format
      const parsedDate = new Date(date);
      if (isNaN(parsedDate.getTime())) {
        res.status(400).json({ 
          success: false, 
          error: 'Invalid date format' 
        });
        return;
      }

      // Verify all drivers exist and have Driver role
      const drivers = await prisma.user.findMany({
        where: {
          id: { in: driverIds },
          role: 'DRIVER'
        }
      });

      if (drivers.length !== driverIds.length) {
        res.status(400).json({ 
          success: false, 
          error: 'One or more users are not valid drivers' 
        });
        return;
      }

      // Create team with members
      const team = await prisma.team.create({
        data: {
          name: name.trim(),
          date: parsedDate,
          createdById: createdById!,
          members: {
            create: driverIds.map(driverId => ({
              userId: driverId
            }))
          }
        },
        include: {
          members: {
            include: {
              user: {
                select: {
                  id: true,
                  name: true,
                  email: true,
                  role: true
                }
              }
            }
          }
        }
      });

      const response: TeamResponse = {
        success: true,
        data: team
      };

      res.status(201).json(response);
    } catch (error) {
      console.error('Error creating team:', error);
      res.status(500).json({ 
        success: false, 
        error: 'Internal server error' 
      });
    }
  }

  static async getTeams(req: Request, res: Response): Promise<void> {
    try {
      const { date } = req.query;

      if (!date || typeof date !== 'string') {
        res.status(400).json({ 
          success: false, 
          error: 'Date parameter is required' 
        });
        return;
      }

      const parsedDate = new Date(date);
      if (isNaN(parsedDate.getTime())) {
        res.status(400).json({ 
          success: false, 
          error: 'Invalid date format' 
        });
        return;
      }

      const teams = await prisma.team.findMany({
        where: {
          date: {
            gte: new Date(parsedDate.toDateString()),
            lt: new Date(new Date(parsedDate.getTime() + 24 * 60 * 60 * 1000).toDateString())
          }
        },
        include: {
          members: {
            include: {
              user: {
                select: {
                  id: true,
                  name: true,
                  email: true,
                  role: true
                }
              }
            }
          }
        },
        orderBy: {
          createdAt: 'desc'
        }
      });

      const response: TeamResponse = {
        success: true,
        data: teams
      };

      res.json(response);
    } catch (error) {
      console.error('Error fetching teams:', error);
      res.status(500).json({ 
        success: false, 
        error: 'Internal server error' 
      });
    }
  }

  static async deleteTeam(req: Request, res: Response): Promise<void> {
    try {
      const { id } = req.params;

      if (!id) {
        res.status(400).json({ 
          success: false, 
          error: 'Team ID is required' 
        });
        return;
      }

      // Check if team has active job assignments
      const activeJobs = await prisma.job.count({
        where: { 
          teamId: id,
          status: { in: ['PENDING', 'IN_PROGRESS'] }
        }
      });

      if (activeJobs > 0) {
        res.status(400).json({ 
          success: false, 
          error: 'Cannot delete team with active job assignments' 
        });
        return;
      }

      // Delete team (cascade will handle members)
      await prisma.team.delete({
        where: { id }
      });

      res.json({ 
        success: true 
      });
    } catch (error) {
      if (error.code === 'P2025') {
        res.status(404).json({ 
          success: false, 
          error: 'Team not found' 
        });
        return;
      }

      console.error('Error deleting team:', error);
      res.status(500).json({ 
        success: false, 
        error: 'Internal server error' 
      });
    }
  }
