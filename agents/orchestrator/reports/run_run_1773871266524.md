**Description:** Implement all auth-related Express routes and middleware. Builds on top of AuthService from auth-core task. Routes handle login, token refresh, and logout. Middleware handles JWT verification for protected routes and RBAC enforcement per role. Refresh tokens are stored and sent via httpOnly cookies only.

**Acceptance Criteria:**
- POST /api/auth/login accepts { email, password }, validates with Zod, returns access token in body and refresh token in httpOnly cookie
- POST /api/auth/refresh reads refresh token from httpOnly cookie, rotates it, returns new access token in body and new refresh token in httpOnly cookie
- POST /api/auth/logout revokes refresh token and clears the httpOnly cookie
- authenticateToken middleware verifies JWT access token from Authorization header, attaches user to req
- requireRole(...roles) middleware checks req.user.role against allowed roles, returns 403 if not allowed
- Drivers are blocked from any route not explicitly allowed for Driver role
- All login errors return generic 401 message — no indication of whether email or password was wrong
- Zod validates all request bodies before they reach route handlers
- Routes registered under /api/auth in Express app

```typescript
import { Router, Response } from 'express';
import { z } from 'zod';
import { AuthService } from '../services/auth.service';
import { AuthenticatedRequest } from '../types/auth.types';

const router = Router();
const authService = new AuthService();

// Zod schemas for validation
const LoginSchema = z.object({
  email: z.string().email().min(1).max(255),
  password: z.string().min(1).max(255)
});

// Cookie configuration
const COOKIE_OPTIONS = {
  httpOnly: true,
  secure: process.env.NODE_ENV === 'production',
  sameSite: 'strict' as const,
  maxAge: 7 * 24 * 60 * 60 * 1000, // 7 days
  path: '/'
};

// POST /api/auth/login
router.post('/login', async (req: AuthenticatedRequest, res: Response) => {
  try {
    // Validate request body with Zod
    const validationResult = LoginSchema.safeParse(req.body);
    if (!validationResult.success) {
      return res.status(400).json({
        error: 'Invalid request format',
        details: validationResult.error.errors
      });
    }

    const { email, password } = validationResult.data;

    // Attempt login
    const result = await authService.login(email, password);
    
    if (!result.success) {
      // Generic error message - don't reveal if email or password was wrong
      return res.status(401).json({
        error: 'Invalid credentials'
      });
    }

    // Set refresh token in httpOnly cookie
    res.cookie('refreshToken', result.refreshToken, COOKIE_OPTIONS);

    // Return access token in response body
    return res.status(200).json({
      accessToken: result.accessToken,
      user: {
        id: result.user.id,
        email: result.user.email,
        role: result.user.role
      }
    });

  } catch (error) {
    console.error('Login error:', error);
    return res.status(500).json({
      error: 'Internal server error'
    });
  }
});

// POST /api/auth/refresh
router.post('/refresh', async (req: AuthenticatedRequest, res: Response) => {
  try {
    const refreshToken = req.cookies?.refreshToken;
    
    if (!refreshToken) {
      return res.status(401).json({
        error: 'Refresh token required'
      });
    }

    // Refresh tokens (this rotates the refresh token)
    const result = await authService.refreshToken(refreshToken);
    
    if (!result.success) {
      // Clear invalid cookie
      res.clearCookie('refreshToken', { path: '/' });
      return res.status(401).json({
        error: 'Invalid refresh token'
      });
    }

    // Set new refresh token in httpOnly cookie
    res.cookie('refreshToken', result.refreshToken, COOKIE_OPTIONS);

    // Return new access token in response body
    return res.status(200).json({
      accessToken: result.accessToken
    });

  } catch (error) {
    console.error('Token refresh error:', error);
    return res.status(500).json({
      error: 'Internal server error'
    });
  }
});

// POST /api/auth/logout
router.post('/logout', async (req: AuthenticatedRequest, res: Response) => {
  try {
    const refreshToken = req.cookies?.refreshToken;
    
    if (refreshToken) {
      // Revoke the refresh token
      await authService.logout(refreshToken);
    }

    // Clear the refresh token cookie
    res.clearCookie('refreshToken', { path: '/' });

    return res.status(200).json({
      message: 'Logged out successfully'
    });

  } catch (error) {
    console.error('Logout error:', error);
    // Still clear the cookie even if logout fails
    res.clearCookie('refreshToken', { path: '/' });
    return res.status(200).json({
      message: 'Logged out successfully'
    });
  }
});

export { router as authRoutes };
```

### backend/src/middleware/authenticate.ts (NEW)
```typescript
import { Response, NextFunction } from 'express';
import jwt from 'jsonwebtoken';
import { AuthService } from '../services/auth.service';
import { AuthenticatedRequest, User } from '../types/auth.types';

const authService = new AuthService();

export const authenticateToken = async (
  req: AuthenticatedRequest, 
  res: Response, 
  next: NextFunction
): Promise<void> => {
  try {
    const authHeader = req.headers.authorization;
    const token = authHeader && authHeader.startsWith('Bearer ') 
      ? authHeader.substring(7) 
      : null;

    if (!token) {
      res.status(401).json({ error: 'Access token required' });
      return;
    }

    // Verify the access token using AuthService
    const result = await authService.verifyAccessToken(token);
    
    if (!result.success || !result.user) {
      res.status(401).json({ error: 'Invalid or expired token' });
      return;
    }

    // Attach user to request
    req.user = result.user;
    next();

  } catch (error) {
    console.error('Token verification error:', error);
    res.status(401).json({ error: 'Token verification failed' });
    return;
  }
};
```

### backend/src/middleware/requireRole.ts (NEW)
```typescript
import { Response, NextFunction } from 'express';
import { AuthenticatedRequest } from '../types/auth.types';

type Role = 'Admin' | 'Manager' | 'Driver';

export const requireRole = (...allowedRoles: Role[]) => {
  return (req: AuthenticatedRequest, res: Response, next: NextFunction): void => {
    try {
      // Ensure user is authenticated (should be handled by authenticateToken middleware first)
      if (!req.user) {
        res.status(401).json({ error: 'Authentication required' });
        return;
      }

      // Check if user's role is in the allowed roles
      if (!allowedRoles.includes(req.user.role)) {
        res.status(403).json({ 
          error: 'Insufficient permissions',
          required: allowedRoles,
          current: req.user.role
        });
        return;
      }

      next();

    } catch (error) {
      console.error('Role verification error:', error);
      res.status(500).json({ error: 'Role verification failed' });
      return;
    }
  };
};

// Convenience function to explicitly block drivers from certain routes
export const blockDrivers = requireRole('Admin', 'Manager');

// Convenience functions for common role requirements
export const adminOnly = requireRole('Admin');
export const managerOrAdmin = requireRole('Admin', 'Manager');
```
