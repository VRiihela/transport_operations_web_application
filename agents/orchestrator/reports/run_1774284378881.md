export const createJobSchema = z.object({
  title: z.string().min(1, 'Title is required').max(255),
  description: z.string().max(1000).optional(),
  scheduledStart: z.string().datetime().nullable(),
  scheduledEnd: z.string().datetime().nullable(),

  assignedDriverId: z.string().uuid().optional(),

  (data) => {
    if (!data.scheduledStart && !data.scheduledEnd && !data.schedulingNote) {
      return false;
    }
    return true;

    message: 'Scheduling note is required when no scheduled times are provided',
    path: ['schedulingNote'],

);

import { Request, Response } from 'express';
import { updateJobSchema } from '../schemas/jobSchemas';
import { Job } from '../models/job';
import { Driver } from '../models/driver';

// Existing controller methods assumed...

export const updateJob = async (req: Request, res: Response): Promise<void> => {
  try {
    const jobId = req.params.id;
    
    // Validate request body
    const validationResult = updateJobSchema.safeParse(req.body);
    if (!validationResult.success) {
      res.status(400).json({
        error: 'Validation error',
        details: validationResult.error.errors
      });
      return;
    }

    const updateData = validationResult.data;

    // Find existing job
    const existingJob = await Job.findOne({
      where: {
        id: jobId,
        deletedAt: null // Ensure not soft-deleted
      }
    });

    if (!existingJob) {
      res.status(404).json({
        error: 'Job not found'
      });
      return;
    }

    // Validate assignedDriverId if provided
    if (updateData.assignedDriverId) {
      const driver = await Driver.findOne({
        where: {
          id: updateData.assignedDriverId,
          isActive: true,
          deletedAt: null
        }
      });

      if (!driver) {
        res.status(400).json({
          error: 'Assigned driver not found or inactive'
        });
        return;
      }
    }

    // Merge with existing data for business rule validation
    const mergedData = {
      scheduledStart: updateData.scheduledStart !== undefined ? updateData.scheduledStart : existingJob.scheduledStart,
      scheduledEnd: updateData.scheduledEnd !== undefined ? updateData.scheduledEnd : existingJob.scheduledEnd,
      schedulingNote: updateData.schedulingNote !== undefined ? updateData.schedulingNote : existingJob.schedulingNote
    };

    // Validate business rule: schedulingNote required if both times are null
    if (!mergedData.scheduledStart && !mergedData.scheduledEnd && !mergedData.schedulingNote) {
      res.status(400).json({
        error: 'Scheduling note is required when no scheduled times are provided'
      });
      return;
    }

    // Update job (explicitly exclude createdById)
    const updatedJob = await existingJob.update({
      title: updateData.title,
      description: updateData.description,
      scheduledStart: updateData.scheduledStart,
      scheduledEnd: updateData.scheduledEnd,
      schedulingNote: updateData.schedulingNote,
      assignedDriverId: updateData.assignedDriverId,
      updatedAt: new Date()
    });

    res.status(200).json({
      data: updatedJob
    });

  } catch (error) {
    console.error('Error updating job:', error);
    res.status(500).json({
      error: 'Internal server error'
    });
  }
};

import { useMutation, useQuery, useQueryClient } from '@tanstack/react-query';
import { updateJob as updateJobApi } from '../api/jobs';
import { UpdateJobInput, Job } from '../types/job';

// Existing hooks assumed...

export const useUpdateJob = () => {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: ({ id, data }: { id: string; data: UpdateJobInput }) => 
      updateJobApi(id, data),
    onSuccess: (updatedJob: Job) => {
      // Optimistic update - update the job in the jobs list
      queryClient.setQueryData(['jobs'], (oldData: Job[] | undefined) => {
        if (!oldData) return oldData;
        return oldData.map(job => 
          job.id === updatedJob.id ? updatedJob : job
        );
      });
    },
  });
};

  job: Job | null;
  drivers: Driver[];
  isOpen: boolean;
  onClose: () => void;
}

export const EditJobModal: React.FC<EditJobModalProps> = ({
  job,
  drivers,
  isOpen,
  onClose,
}) => {
  const [formData, setFormData] = useState<UpdateJobInput>({});
  const [error, setError] = useState<string>('');
  
  const updateJobMutation = useUpdateJob();

  useEffect(() => {
    if (job) {
      setFormData({
        title: job.title,
        description: job.description || '',
        scheduledStart: job.scheduledStart,
        scheduledEnd: job.scheduledEnd,
        schedulingNote: job.schedulingNote || '',
        assignedDriverId: job.assignedDriverId || '',
      });
      setError('');
    }

  const handleInputChange = (
    e: React.ChangeEvent<HTMLInputElement | HTMLTextAreaElement | HTMLSelectElement>
  ) => {
    const { name, value } = e.target;
    setFormData(prev => ({
      ...prev,
      [name]: value === '' ? undefined : value
    }));
  };

  const handleDateTimeChange = (name: string, value: string) => {
    setFormData(prev => ({
      ...prev,
      [name]: value === '' ? null : value
    }));

    e.preventDefault();
    if (!job) return;

      await updateJobMutation.mutateAsync({

        data: formData

      onClose();

      setError(err instanceof Error ? err.message : 'Failed to update job');
