    res.json({ data: jobs });

// New endpoint for updating driver notes
export const updateDriverNotes = async (req: AuthenticatedRequest, res: Response): Promise<void> => {
  try {
    const jobId = req.params.id;
    const userId = req.user?.id;
    const { driverNotes } = req.body as UpdateDriverNotesRequest;

    // Validate input
    if (typeof driverNotes !== 'string') {
      res.status(400).json({ error: 'driverNotes must be a string' });
      return;
    }

    if (driverNotes.length > 1000) {
      res.status(400).json({ error: 'driverNotes must not exceed 1000 characters' });
      return;
    }

    // Check if job exists
    const job = await jobService.getJobById(jobId);
    if (!job) {
      res.status(404).json({ error: 'Job not found' });
      return;
    }

    // Verify user is the assigned driver
    if (job.assignedDriverId !== userId) {
      res.status(403).json({ error: 'Only the assigned driver can update notes for this job' });
      return;
    }

    // Sanitize input to prevent XSS
    const sanitizedNotes = driverNotes.trim();

    // Update driver notes
    const updatedJob = await jobService.updateDriverNotes(jobId, sanitizedNotes);
    
    res.json({ data: updatedJob });
  } catch (error) {
    console.error('Error updating driver notes:', error);
    res.status(500).json({ error: 'Failed to update driver notes' });
  }
};

export const updateJobDriverNotes = async (
  jobId: string, 
  driverNotes: string
): Promise<UpdateDriverNotesResponse> => {
  const response = await fetch(`/api/jobs/${jobId}/notes`, {
    method: 'PATCH',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${getAuthToken()}` // Use existing auth pattern
    },
    body: JSON.stringify({ driverNotes } as UpdateDriverNotesRequest)
  });

  if (!response.ok) {
    const errorData = await response.json();
    throw new Error(errorData.error || 'Failed to update driver notes');
  }

  return response.json() as Promise<UpdateDriverNotesResponse>;
};

import React, { useState, useCallback } from 'react';
import { Job, JobStatus } from '../types/Job';
import { updateJobDriverNotes } from '../services/jobService';

interface JobRowProps {
  job: Job;
  onNotesUpdate: (jobId: string, updatedJob: Job) => void;
}

const JobRow: React.FC<JobRowProps> = ({ job, onNotesUpdate }) => {
  const [notes, setNotes] = useState(job.driverNotes || '');
  const [isSaving, setIsSaving] = useState(false);
  const [saveStatus, setSaveStatus] = useState<'idle' | 'success' | 'error'>('idle');
  const [errorMessage, setErrorMessage] = useState('');

  const canEdit = job.status === JobStatus.ASSIGNED || job.status === JobStatus.IN_PROGRESS;

  const handleSave = useCallback(async () => {
    if (notes.length > 1000) {
      setSaveStatus('error');
      setErrorMessage('Notes must not exceed 1000 characters');
      return;
    }

    setIsSaving(true);
    setSaveStatus('idle');
    setErrorMessage('');

    try {
      const response = await updateJobDriverNotes(job.id, notes);
      onNotesUpdate(job.id, response.data);
      setSaveStatus('success');
      setTimeout(() => setSaveStatus('idle'), 2000);
    } catch (error) {
      setSaveStatus('error');
      setErrorMessage(error instanceof Error ? error.message : 'Failed to save notes');
    } finally {
      setIsSaving(false);
    }
  }, [job.id, notes, onNotesUpdate]);

  const handleNotesChange = (e: React.ChangeEvent<HTMLTextAreaElement>) => {
    setNotes(e.target.value);
    if (saveStatus !== 'idle') {
      setSaveStatus('idle');
      setErrorMessage('');
    }
  };

  return (
    <div className="job-row">
      {/* Existing job row content */}
      <div className="job-details">
        <h3>{job.title}</h3>
        <p>{job.description}</p>
        <span className={`status ${job.status.toLowerCase()}`}>
          {job.status}
        </span>
      </div>

      {/* Driver notes section */}
      <div className="driver-notes-section">
        <label htmlFor={`notes-${job.id}`} className="notes-label">
          Driver Notes:
        </label>
        <div className="notes-input-container">
          <textarea
            id={`notes-${job.id}`}
            value={notes}
            onChange={handleNotesChange}
            disabled={!canEdit || isSaving}
            placeholder={canEdit ? "Add notes or deviation info..." : ""}
            className={`notes-textarea ${!canEdit ? 'read-only' : ''}`}
            maxLength={1000}
            rows={3}
          />
          <div className="notes-footer">
            <span className="char-count">
              {notes.length}/1000
            </span>
            {canEdit && (
              <button
                type="button"
                onClick={handleSave}
                disabled={isSaving || notes === (job.driverNotes || '')}
                className="save-notes-btn"
              >
                {isSaving ? 'Saving...' : 'Save'}
              </button>
            )}
          </div>
        </div>
        
        {/* Status messages */}
        {saveStatus === 'success' && (
          <div className="save-status success">Saved</div>
        )}
        {saveStatus === 'error' && (
          <div className="save-status error">{errorMessage}</div>
        )}
      </div>
    </div>
  );
};

// Update main component to handle notes updates
export const MyJobsPage: React.FC = () => {

  const handleNotesUpdate = useCallback((jobId: string, updatedJob: Job) => {

      prevJobs.map(job => job.id === jobId ? updatedJob : job)
    );
  }, []);

  // ... existing logic for fetching jobs

  return (
    <div className="my-jobs-page">
      <h1>My Jobs</h1>
      {jobs.map(job => (
