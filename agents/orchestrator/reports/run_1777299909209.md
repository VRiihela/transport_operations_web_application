```typescript
import { Job } from '../types/job';
import { User } from '../types/user';

const API_BASE = process.env.REACT_APP_API_BASE_URL || '/api';

interface ApiResponse<T> {
  success: boolean;
  data: T;
  message?: string;
}

interface ApiError {
  success: false;
  message: string;
  code?: string;
}

class ApiService {
  private async request<T>(
    endpoint: string,
    options: RequestInit = {}
  ): Promise<T> {
    const url = `${API_BASE}${endpoint}`;
    
    const config: RequestInit = {
      headers: {
        'Content-Type': 'application/json',
        ...options.headers,
      },
      ...options,
    };

    try {
      const response = await fetch(url, config);
      
      if (!response.ok) {
        throw new Error(`HTTP ${response.status}: ${response.statusText}`);
      }

      const result: ApiResponse<T> | ApiError = await response.json();
      
      if (!result.success) {
        throw new Error(result.message || 'API request failed');
      }

      return (result as ApiResponse<T>).data;
    } catch (error) {
      console.error(`API request failed for ${endpoint}:`, error);
      throw error;
    }
  }

  async getJobs(): Promise<Job[]> {
    return this.request<Job[]>('/jobs?status=DRAFT&status=ASSIGNED');
  }

  async getDrivers(): Promise<User[]> {
    return this.request<User[]>('/users?role=DRIVER');
  }

  async updateJobAssignment(
    jobId: string,
    assignedDriverId: string | null
  ): Promise<Job> {
    return this.request<Job>(`/jobs/${jobId}`, {
      method: 'PATCH',
      body: JSON.stringify({ assignedDriverId }),
    });
  }
}

export const apiService = new ApiService();
```

### src/components/JobBoard.tsx

import React, { useState, useEffect, useCallback } from 'react';
import { DragDropContext, DropResult } from 'react-beautiful-dnd';
import { Job } from '../types/job';
import { User } from '../types/user';
import { apiService } from '../services/api';

interface JobBoardState {
  jobs: Job[];
  drivers: User[];
}

interface LoadingState {
  jobs: boolean;
  drivers: boolean;
  updating: boolean;
}

interface ErrorState {
  jobs: string | null;
  drivers: string | null;
  update: string | null;
}

const JobBoard: React.FC = () => {
  const [state, setState] = useState<JobBoardState>({
    jobs: [],
    drivers: [],
  });

  const [loading, setLoading] = useState<LoadingState>({
    jobs: true,
    drivers: true,
    updating: false,
  });

  const [errors, setErrors] = useState<ErrorState>({
    jobs: null,
    drivers: null,
    update: null,
  });

  // Store previous state for rollback
  const [stateSnapshot, setStateSnapshot] = useState<JobBoardState | null>(null);

  const loadJobs = useCallback(async (): Promise<void> => {
    try {
      setLoading(prev => ({ ...prev, jobs: true }));
      setErrors(prev => ({ ...prev, jobs: null }));
      
      const jobs = await apiService.getJobs();
      setState(prev => ({ ...prev, jobs }));
    } catch (error) {
      const message = error instanceof Error ? error.message : 'Failed to load jobs';
      setErrors(prev => ({ ...prev, jobs: message }));
      console.error('Failed to load jobs:', error);
    } finally {
      setLoading(prev => ({ ...prev, jobs: false }));
    }
  }, []);

  const loadDrivers = useCallback(async (): Promise<void> => {
    try {
      setLoading(prev => ({ ...prev, drivers: true }));
      setErrors(prev => ({ ...prev, drivers: null }));
      
      const drivers = await apiService.getDrivers();
      setState(prev => ({ ...prev, drivers }));
    } catch (error) {
      const message = error instanceof Error ? error.message : 'Failed to load drivers';
      setErrors(prev => ({ ...prev, drivers: message }));
      console.error('Failed to load drivers:', error);
    } finally {
      setLoading(prev => ({ ...prev, drivers: false }));
    }
  }, []);

  useEffect(() => {
    void loadJobs();
    void loadDrivers();
  }, [loadJobs, loadDrivers]);

  const onDragEnd = useCallback(async (result: DropResult): Promise<void> => {
    const { destination, source, draggableId } = result;

    // No destination or same position
    if (!destination || 
        (destination.droppableId === source.droppableId && 
         destination.index === source.index)) {
      return;
    }

    // Find the job being moved
    const jobId = draggableId;
    const job = state.jobs.find(j => j.id === jobId);
    
    if (!job) {
      console.error('Job not found for drag operation:', jobId);
      return;
    }

    // Take snapshot for potential rollback
    setStateSnapshot({ ...state });

    // Determine new assignment
    const newAssignedDriverId = destination.droppableId === 'unassigned' 
      ? null 
      : destination.droppableId;

    // Optimistically update state
    setState(prev => ({
      ...prev,
      jobs: prev.jobs.map(j => 
        j.id === jobId 
          ? { ...j, assignedDriverId: newAssignedDriverId || undefined }
          : j
      ),
    }));

    // Update backend
    try {
      setLoading(prev => ({ ...prev, updating: true }));
      setErrors(prev => ({ ...prev, update: null }));

      await apiService.updateJobAssignment(jobId, newAssignedDriverId);
      
      // Clear snapshot on success
      setStateSnapshot(null);
    } catch (error) {
      // Rollback on failure
      if (stateSnapshot) {
        setState(stateSnapshot);
        setStateSnapshot(null);
      }
      
      const message = error instanceof Error ? error.message : 'Failed to update assignment';
      setErrors(prev => ({ ...prev, update: message }));
      console.error('Failed to update job assignment:', error);
    } finally {
      setLoading(prev => ({ ...prev, updating: false }));
    }
  }, [state, stateSnapshot]);

  const isInitialLoading = loading.jobs || loading.drivers;

  if (isInitialLoading) {
    return (
      <div className="job-board-loading">
        <div className="spinner" />
        <p>Loading job assignments...</p>
      </div>
    );
  }

  if (errors.jobs || errors.drivers) {
    return (
      <div className="job-board-error">
        <h3>Failed to load data</h3>
        {errors.jobs && <p>Jobs: {errors.jobs}</p>}
        {errors.drivers && <p>Drivers: {errors.drivers}</p>}
        <button onClick={() => {
          void loadJobs();
          void loadDrivers();
        }}>
          Retry
        </button>
      </div>
    );
  }

  return (
    <div className={`job-board ${loading.updating ? 'updating' : ''}`}>
      {errors.update && (
        <div className="error-banner">
          <p>Update failed: {errors.update}</p>
          <button onClick={() => setErrors(prev => ({ ...prev, update: null }))}>
            Dismiss
          </button>
        </div>
      )}
      
      <DragDropContext onDragEnd={onDragEnd}>
        {/* Render your drag-and-drop columns here */}
        {/* This would include your existing drag-and-drop UI components */}
        {/* using state.jobs and state.drivers instead of mock data */}
      </DragDropContext>
      
      {loading.updating && (
        <div className="updating-overlay">
          <div className="spinner" />
          <p>Saving changes...</p>
        </div>
      )}
    </div>
  );
};

export default JobBoard;
