  max: 10, // 10 requests per window per IP
  message: {
    error: 'Too many login attempts',
    message: 'Maximum of 10 login attempts allowed per 15 minutes. Please try again later.',
    retryAfter: '15 minutes'
  },
  standardHeaders: true, // Return rate limit info in headers
  legacyHeaders: false,
  handler: (req: Request, res: Response) => {

         error: 'Too many login attempts',
         message: 'Maximum of 10 login attempts allowed per 15 minutes. Please try again later.',

import { PrismaClient } from '@prisma/client';
import type { Request } from 'express';
import type { AuditEvent, AuditLogData, SafeIP } from '../types/audit.types';

const prisma = new PrismaClient();

export class AuditService {
  /**
   * Extract and validate IP address from request
   * Handles X-Forwarded-For header with validation
   */
  private static extractSafeIP(req: Request): SafeIP {
    const forwardedFor = req.headers['x-forwarded-for'];
    const connectionIP = req.socket.remoteAddress || 'unknown';

    // If no X-Forwarded-For header, use connection IP
    if (!forwardedFor) {
      return { ip: connectionIP, trusted: true };
    }

    // Handle X-Forwarded-For header (can be comma-separated list)
    const forwardedIPs = Array.isArray(forwardedFor) 
      ? forwardedFor[0] 
      : forwardedFor;

    const firstIP = forwardedIPs.split(',')[0]?.trim();

    // Validate IP format (basic IPv4/IPv6 check)
    const ipv4Regex = /^(\d{1,3}\.){3}\d{1,3}$/;
    const ipv6Regex = /^([0-9a-fA-F]{1,4}:){7}[0-9a-fA-F]{1,4}$|^::1$|^::$/;

    if (firstIP && (ipv4Regex.test(firstIP) || ipv6Regex.test(firstIP))) {
      return { ip: firstIP, trusted: false }; // X-Forwarded-For is less trusted
    }

    // Fallback to connection IP if forwarded IP is invalid
    return { ip: connectionIP, trusted: true };
  }

  /**
   * Get User-Agent header safely
   */
  private static extractUserAgent(req: Request): string | undefined {
    const userAgent = req.headers['user-agent'];
    
    // Truncate very long user agent strings to prevent abuse
    if (typeof userAgent === 'string') {
      return userAgent.length > 500 ? userAgent.substring(0, 500) : userAgent;
    }
    
    return undefined;
  }

  /**
   * Log audit event to database
   * This should not throw errors that break the auth flow
   */
  public static async logEvent(data: AuditLogData): Promise<void> {
    try {
      await prisma.auditLog.create({
        data: {
          event: data.event,
          userId: data.userId,
          ip: data.ip,
          userAgent: data.userAgent,
          metadata: data.metadata ? JSON.stringify(data.metadata) : null
        }
      });
    } catch (error) {
      // Log to stdout but don't throw - audit failures shouldn't break auth
      console.error('Audit logging failed:', error);
    }
  }

  /**
   * Helper to log auth events from Express requests
   */
  public static async logAuthEvent(
    req: Request, 
    event: AuditEvent, 
    userId?: number,
    metadata?: Record<string, unknown>
  ): Promise<void> {
    const { ip } = this.extractSafeIP(req);
    const userAgent = this.extractUserAgent(req);

    await this.logEvent({
      event,
      userId,
      ip,
      userAgent,
      metadata
    });
  }

  /**
   * Log login success
   */
  public static async logLoginSuccess(req: Request, userId: number): Promise<void> {
    await this.logAuthEvent(req, 'LOGIN_SUCCESS', userId);
  }

  /**
   * Log login failure - includes attempted email for ops visibility
   * NEVER logs password
   */
  public static async logLoginFailure(req: Request, email: string): Promise<void> {
    await this.logAuthEvent(req, 'LOGIN_FAILURE', undefined, { 
      attemptedEmail: email 
    });
  }

  /**
   * Log logout event
   */
  public static async logLogout(req: Request, userId: number): Promise<void> {
    await this.logAuthEvent(req, 'LOGOUT', userId);
  }

  /**
   * Log token refresh
   */
  public static async logTokenRefresh(req: Request, userId: number): Promise<void> {
    await this.logAuthEvent(req, 'TOKEN_REFRESH', userId);
  }

  /**
   * Log token reuse detection - HIGH severity security event
   */
  public static async logTokenReuseDetected(
    req: Request, 
    userId: number, 
    tokenId: string
  ): Promise<void> {
    await this.logAuthEvent(req, 'TOKEN_REUSE_DETECTED', userId, {
      severity: 'HIGH',
      tokenId: tokenId // Safe to log token ID for investigation
    });
  }
}

import { PrismaClient } from '@prisma/client';

const prisma = new PrismaClient();

export interface CleanupResult {
  deletedCount: number;
  errors: string[];
}

/**
 * Clean up expired refresh tokens
 * Only deletes tokens that are both expired AND revoked
 * Safe to run repeatedly (idempotent)
 */
export async function cleanupExpiredTokens(): Promise<CleanupResult> {
  const result: CleanupResult = {
    deletedCount: 0,
    errors: []
  };

  try {
    console.log('Starting expired token cleanup...');
    
    const now = new Date();
    
    // Delete tokens that are both expired AND revoked
    // This is a safety measure - we only clean up tokens that are:
    // 1. Past their expiration date
    // 2. Already marked as revoked
    const deleteResult = await prisma.refreshToken.deleteMany({
      where: {
        AND: [
          { expiresAt: { lt: now } },      // Expired
          { revokedAt: { not: null } }     // Revoked
        ]
      }
    });

    result.deletedCount = deleteResult.count;
    
    console.log(`Cleanup completed: ${result.deletedCount} expired tokens removed`);
    
  } catch (error) {
    const errorMessage = error instanceof Error ? error.message : 'Unknown error';
    result.errors.push(errorMessage);
    console.error('Token cleanup failed:', errorMessage);
  }

  return result;
}

/**
 * Get count of tokens that would be cleaned up (dry run)
 */
export async function getExpiredTokenCount(): Promise<number> {
  try {
    const now = new Date();
    
    const count = await prisma.refreshToken.count({
      where: {
        AND: [
          { expiresAt: { lt: now } },
          { revokedAt: { not: null } }
        ]
      }
    });
    
    return count;
  } catch (error) {
    console.error('Failed to count expired tokens:', error);
    return 0;
  }
}

// CLI execution support
if (require.main === module) {
  cleanupExpiredTokens()
    .then((result) => {
      console.log('Cleanup result:', result);
      process.exit(result.errors.length > 0 ? 1 : 0);
    })
    .catch((error) => {
      console.error('Cleanup script failed:', error);
      process.exit(1);
    });
}
