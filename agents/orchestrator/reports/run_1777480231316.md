```typescript
import { Job, DriverJobsQuery } from '../types/job.types';
import { isValid, parseISO, startOfDay, endOfDay } from 'date-fns';

export class JobService {
  async getDriverJobsByDate(driverId: string, query: DriverJobsQuery): Promise<Job[]> {
    // Validate date parameter
    const parsedDate = parseISO(query.date);
    if (!isValid(parsedDate)) {
      throw new Error('Invalid date format. Expected YYYY-MM-DD');
    }

    const startDate = startOfDay(parsedDate);
    const endDate = endOfDay(parsedDate);

    // Database query with proper filtering and sorting
    const jobs = await this.jobRepository.find({
      where: {
        assignedDriverId: driverId,
        OR: [
          {
            scheduledStart: {
              gte: startDate,
              lte: endDate,
            },
          },
          {
            // Include jobs without scheduledStart for the driver
            scheduledStart: null,
            createdAt: {
              gte: startDate,
              lte: endDate,
            },
          },
        ],
      },
      orderBy: [
        {
          scheduledStart: {
            sort: 'asc',
            nulls: 'last',
          },
        },
      ],
    });

    return jobs;
  }
}
```

```typescript
import { Request, Response } from 'express';
import { JobService } from '../services/jobService';
import { DriverJobsQuery } from '../types/job.types';
import { z } from 'zod';

const dateQuerySchema = z.object({
  date: z.string().regex(/^\d{4}-\d{2}-\d{2}$/, 'Date must be in YYYY-MM-DD format'),
});

export class DriverJobsController {
  constructor(private jobService: JobService) {}

  async getMyJobs(req: Request, res: Response): Promise<void> {
    try {
      // Validate query parameters
      const validatedQuery = dateQuerySchema.parse(req.query);
      
      // Get driver ID from authenticated session
      const driverId = req.user?.id;
      if (!driverId) {
        res.status(401).json({ error: 'Unauthorized' });
        return;
      }

      const jobs = await this.jobService.getDriverJobsByDate(driverId, validatedQuery);

      res.json({
        jobs,
        date: validatedQuery.date,
        total: jobs.length,
      });
    } catch (error) {
      if (error instanceof z.ZodError) {
        res.status(400).json({ error: 'Invalid query parameters', details: error.errors });
        return;
      }
      
      console.error('Error fetching driver jobs:', error);
      res.status(500).json({ error: 'Internal server error' });
    }
  }
}
```

```typescript
import { format, parseISO } from 'date-fns';
import { fi } from 'date-fns/locale';

export const formatFinnishDateTime = (date: Date | string | null): string => {
  if (!date) return '';
  
  const parsedDate = typeof date === 'string' ? parseISO(date) : date;
  return format(parsedDate, 'dd.MM.yyyy HH:mm', { locale: fi });
};

export const formatFinnishDate = (date: Date | string): string => {
  const parsedDate = typeof date === 'string' ? parseISO(date) : date;
  return format(parsedDate, 'dd.MM.yyyy', { locale: fi });
};

export const formatISODate = (date: Date): string => {
  return format(date, 'yyyy-MM-dd');
};

export const formatTimeRange = (start: Date | string | null, end: Date | string | null): string => {
  if (!start && !end) return '';
  if (!start) return `– ${formatFinnishDateTime(end)}`;
  if (!end) return formatFinnishDateTime(start);
  
  const startFormatted = formatFinnishDateTime(start);
  const endTime = format(typeof end === 'string' ? parseISO(end) : end, 'HH:mm');
  
  return `${startFormatted}–${endTime}`;
};
```

```typescript
import { DriverJobsResponse } from '../types/job.types';

class JobApiService {
  private baseUrl = process.env.REACT_APP_API_BASE_URL || '/api';

  async getDriverJobs(date: string): Promise<DriverJobsResponse> {
    try {
      const response = await fetch(
        `${this.baseUrl}/driver/my-jobs?date=${encodeURIComponent(date)}`,
        {
          method: 'GET',
          headers: {
            'Content-Type': 'application/json',
          },
          credentials: 'include', // Include cookies for authentication
        }
      );

      if (!response.ok) {
        if (response.status === 401) {
          throw new Error('Unauthorized');
        }
        if (response.status === 400) {
          const errorData = await response.json();
          throw new Error(errorData.error || 'Invalid request');
        }
        throw new Error('Failed to fetch jobs');
      }

      return await response.json();
    } catch (error) {
      console.error('API Error:', error);
      throw error;
    }
  }
}

export const jobApi = new JobApiService();
```

```typescript
import React, { useState, useEffect, useCallback } from 'react';
import { DatePicker } from '../../components/DatePicker';
import { JobCard } from '../../components/JobCard';
import { jobApi } from '../../services/jobApi';
import { Job } from '../../types/job.types';
import { formatISODate } from '../../utils/dateFormatters';
import { debounce } from 'lodash';

interface MyJobsPageState {
  jobs: Job[];
  selectedDate: Date;
  loading: boolean;
  error: string | null;
}

export const MyJobsPage: React.FC = () => {
  const [state, setState] = useState<MyJobsPageState>({
    jobs: [],
    selectedDate: new Date(),
    loading: true,
    error: null,
  });

  const fetchJobs = useCallback(async (date: Date) => {
    setState(prev => ({ ...prev, loading: true, error: null }));
    
    try {
      const isoDate = formatISODate(date);
      const response = await jobApi.getDriverJobs(isoDate);
      
      setState(prev => ({
        ...prev,
        jobs: response.jobs,
        loading: false,
      }));
    } catch (error) {
      setState(prev => ({
        ...prev,
        jobs: [],
        loading: false,
        error: error instanceof Error ? error.message : 'Unknown error occurred',
      }));
    }
  }, []);

  // Debounced version to prevent excessive API calls
  const debouncedFetchJobs = useCallback(
    debounce((date: Date) => fetchJobs(date), 300),
    [fetchJobs]
  );

  useEffect(() => {
    fetchJobs(state.selectedDate);
  }, []); // Only run on mount

  const handleDateChange = useCallback((date: Date) => {
    setState(prev => ({ ...prev, selectedDate: date }));
    debouncedFetchJobs(date);
  }, [debouncedFetchJobs]);

  const renderEmptyState = () => (
    <div className="empty-state">
      <div className="empty-state__icon">📅</div>
      <h3 className="empty-state__title">Ei töitä valitulle päivälle</h3>
      <p className="empty-state__message">
        Valitse toinen päivämäärä nähdäksesi työsi.
      </p>
    </div>
  );

  const renderError = () => (
    <div className="error-state">
      <div className="error-state__icon">⚠️</div>
      <h3 className="error-state__title">Virhe ladattaessa töitä</h3>
      <p className="error-state__message">{state.error}</p>
      <button 
        className="error-state__retry"
        onClick={() => fetchJobs(state.selectedDate)}
      >
        Yritä uudelleen
      </button>
    </div>
  );

  const renderLoading = () => (
    <div className="loading-state">
      <div className="loading-spinner" />
      <p>Ladataan töitä...</p>
    </div>
  );

  return (
    <div className="my-jobs-page">
      <div className="my-jobs-page__header">
        <h1>Omat työt</h1>
        <DatePicker
          selectedDate={state.selectedDate}
          onChange={handleDateChange}
          disabled={state.loading}
          className="my-jobs-page__date-picker"
        />
      </div>

      <div className="my-jobs-page__content">
        {state.loading && renderLoading()}
        {state.error && !state.loading && renderError()}
        {!state.loading && !state.error && state.jobs.length === 0 && renderEmptyState()}
        {!state.loading && !state.error && state.jobs.length > 0 && (
          <div className="jobs-list">
            {state.jobs.map(job => (
              <JobCard key={job.id} job={job} />
            ))}
          </div>
        )}
      </div>
    </div>
  );
};
```
