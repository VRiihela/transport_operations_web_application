```typescript
import { z } from 'zod';
import { Request, Response } from 'express';
import { PrismaClient, JobStatus } from '@prisma/client';

const prisma = new PrismaClient();

// Validation schema for adding team member
const addTeamMemberSchema = z.object({
  userId: z.string().uuid(),
  role: z.enum(['DRIVER', 'LEAD'])
});

// POST /api/teams/:id/members - Add team member with job reassignment
export async function addTeamMember(req: Request, res: Response) {
  try {
    const teamId = z.string().uuid().parse(req.params.id);
    const { userId, role } = addTeamMemberSchema.parse(req.body);

    // Verify team exists and user exists
    const [team, user] = await Promise.all([
      prisma.team.findUnique({ where: { id: teamId } }),
      prisma.user.findUnique({ where: { id: userId } })
    ]);

    if (!team) {
      return res.status(404).json({ error: 'Team not found' });
    }

    if (!user) {
      return res.status(404).json({ error: 'User not found' });
    }

    // Check if user is already a team member
    const existingMembership = await prisma.teamMember.findFirst({
      where: { userId }
    });

    if (existingMembership) {
      return res.status(409).json({ error: 'User is already a team member' });
    }

    // Transaction: Add team member and reassign driver jobs if role is DRIVER
    const result = await prisma.$transaction(async (tx) => {
      // Add team member
      const teamMember = await tx.teamMember.create({
        data: {
          teamId,
          userId,
          role,
          joinedAt: new Date()
        }
      });

      let reassignedJobsCount = 0;

      // If adding a driver, reassign their ASSIGNED/IN_PROGRESS jobs to the team
      if (role === 'DRIVER') {
        const jobUpdateResult = await tx.job.updateMany({
          where: {
            assignedDriverId: userId,
            status: {
              in: ['ASSIGNED', 'IN_PROGRESS']
            }
          },
          data: {
            assignedDriverId: null,
            teamId: teamId
          }
        });
        
        reassignedJobsCount = jobUpdateResult.count;
      }

      return { teamMember, reassignedJobsCount };
    });

    res.status(201).json({
      teamMember: result.teamMember,
      reassignedJobsCount: result.reassignedJobsCount
    });

  } catch (error) {
    if (error instanceof z.ZodError) {
      return res.status(400).json({ error: 'Invalid input', details: error.errors });
    }
    console.error('Error adding team member:', error);
    res.status(500).json({ error: 'Internal server error' });
  }
}

// DELETE /api/teams/:id - Delete team with job cleanup
export async function deleteTeam(req: Request, res: Response) {
  try {
    const teamId = z.string().uuid().parse(req.params.id);

    // Check if team exists
    const team = await prisma.team.findUnique({ where: { id: teamId } });
    
    if (!team) {
      return res.status(404).json({ error: 'Team not found' });
    }

    // Check for active team jobs (ASSIGNED or IN_PROGRESS)
    const activeJobs = await prisma.job.findMany({
      where: {
        teamId,
        status: {
          in: ['ASSIGNED', 'IN_PROGRESS']
        }
      },
      select: { id: true }
    });

    if (activeJobs.length > 0) {
      return res.status(409).json({ 
        error: 'Cannot delete team with active jobs',
        activeJobsCount: activeJobs.length 
      });
    }

    // Transaction: Clear team jobs and delete team
    const result = await prisma.$transaction(async (tx) => {
      // Set all team jobs to DRAFT and clear teamId
      const jobUpdateResult = await tx.job.updateMany({
        where: { teamId },
        data: {
          status: 'DRAFT',
          teamId: null
        }
      });

      // Delete team members first (due to foreign key constraint)
      await tx.teamMember.deleteMany({
        where: { teamId }
      });

      // Delete the team
      await tx.team.delete({
        where: { id: teamId }
      });

      return { clearedJobsCount: jobUpdateResult.count };
    });

    res.json({
      message: 'Team deleted successfully',
      deletedTeamId: teamId,
      clearedJobsCount: result.clearedJobsCount
    });

  } catch (error) {
    if (error instanceof z.ZodError) {
      return res.status(400).json({ error: 'Invalid team ID' });
    }
    console.error('Error deleting team:', error);
    res.status(500).json({ error: 'Internal server error' });
  }
}
```

```typescript
// Add to existing job update logic
const updateJobSchema = z.object({
  title: z.string().optional(),
  description: z.string().optional(),
  assignedDriverId: z.string().uuid().nullable().optional(),
  teamId: z.string().uuid().nullable().optional(),
  status: z.enum(['DRAFT', 'ASSIGNED', 'IN_PROGRESS', 'COMPLETED', 'CANCELLED']).optional(),
  priority: z.enum(['LOW', 'MEDIUM', 'HIGH', 'URGENT']).optional(),
  scheduledDate: z.string().datetime().optional(),
  location: z.string().optional()
}).refine(data => {
  // Mutual exclusivity: cannot assign both driver and team
  return !(data.assignedDriverId && data.teamId);
}, {
  message: "Job cannot be assigned to both a driver and a team"
});

export async function updateJob(req: Request, res: Response) {
  try {
    const jobId = z.string().uuid().parse(req.params.id);
    const updateData = updateJobSchema.parse(req.body);

    // Verify job exists
    const existingJob = await prisma.job.findUnique({ where: { id: jobId } });
    
    if (!existingJob) {
      return res.status(404).json({ error: 'Job not found' });
    }

    // Validate team exists if teamId provided
    if (updateData.teamId) {
      const team = await prisma.team.findUnique({ where: { id: updateData.teamId } });
      if (!team) {
        return res.status(404).json({ error: 'Team not found' });
      }
    }

    // Validate driver exists if assignedDriverId provided
    if (updateData.assignedDriverId) {
      const driver = await prisma.user.findUnique({ where: { id: updateData.assignedDriverId } });
      if (!driver) {
        return res.status(404).json({ error: 'Driver not found' });
      }
    }

    // Update job with mutual exclusivity enforcement
    const updatedJob = await prisma.job.update({
      where: { id: jobId },
      data: {
        ...updateData,
        // Ensure mutual exclusivity
        assignedDriverId: updateData.teamId ? null : updateData.assignedDriverId,
        teamId: updateData.assignedDriverId ? null : updateData.teamId,
        updatedAt: new Date()
      },
      include: {
        assignedDriver: {
          select: { id: true, firstName: true, lastName: true, email: true }
        },
        team: {
          select: { id: true, name: true }
        },
        createdBy: {
          select: { id: true, firstName: true, lastName: true, email: true }
        }
      }
    });

    res.json(updatedJob);

  } catch (error) {
    if (error instanceof z.ZodError) {
      return res.status(400).json({ error: 'Invalid input', details: error.errors });
    }
    console.error('Error updating job:', error);
    res.status(500).json({ error: 'Internal server error' });
  }
}
```

```typescript
import { useState, useEffect } from 'react';
import { TeamOption } from '../types/api';

export function useTeams() {
  const [teams, setTeams] = useState<TeamOption[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    async function fetchTeams() {
      try {
        const response = await fetch('/api/teams');
        if (!response.ok) {
          throw new Error('Failed to fetch teams');
        }
        const data = await response.json();
        setTeams(data.map((team: any) => ({
          id: team.id,
          name: team.name
        })));
      } catch (err) {
        setError(err instanceof Error ? err.message : 'Unknown error');
      } finally {
        setLoading(false);
      }
    }

    fetchTeams();
  }, []);

  return { teams, loading, error };
}
```

### 6. Frontend Job Assignment Component (src/components/JobAssignment.tsx)

```typescript
import React from 'react';
import { JobAssignmentProps, DriverOption, TeamOption } from '../types/api';
import { useTeams } from '../hooks/useTeams';
import { useDrivers } from '../hooks/useDrivers';
import styles from './JobAssignment.module.css';

export function JobAssignment({
  currentDriverId,
  currentTeamId,
  onDriverChange,
  onTeamChange
}: JobAssignmentProps) {
  const { teams, loading: teamsLoading } = useTeams();
  const { drivers, loading: driversLoading } = useDrivers();

  const handleAssignmentTypeChange = (type: 'driver' | 'team' | 'none') => {
    if (type === 'driver') {
      onTeamChange(null);
    } else if (type === 'team') {
      onDriverChange(null);
    } else {
      onDriverChange(null);
      onTeamChange(null);
    }
  };

  const assignmentType = currentDriverId ? 'driver' : currentTeamId ? 'team' : 'none';

  return (
    <div className={styles.jobAssignment}>
      <div className={styles.assignmentTypeSelector}>
        <label className={styles.label}>Assignment Type:</label>
        <div className={styles.radioGroup}>
          <label className={styles.radioLabel}>
            <input
              type="radio"
              name="assignmentType"
              value="none"
              checked={assignmentType === 'none'}
              onChange={() => handleAssignmentTypeChange('none')}
            />
            Unassigned
          </label>
          <label className={styles.radioLabel}>
            <input
              type="radio"
              name="assignmentType"
              value="driver"
              checked={assignmentType === 'driver'}
              onChange={() => handleAssignmentTypeChange('driver')}
            />
            Individual Driver
          </label>
          <label className={styles.radioLabel}>
            <input
              type="radio"
              name="assignmentType"
              value="team"
              checked={assignmentType === 'team'}
              onChange={() => handleAssignmentTypeChange('team')}
            />
            Team
          </label>
        </div>
      </div>

      {assignmentType === 'driver' && (
        <div className={styles.driverSelector}>
          <label htmlFor="driverSelect" className={styles.label}>
            Select Driver:
          </label>
          <select
            id="driverSelect"
            value={currentDriverId || ''}
            onChange={(e) => onDriverChange(e.target.value || null)}
            className={styles.select}
            disabled={driversLoading}
          >
            <option value="">Select a driver...</option>
            {drivers.map((driver) => (
              <option key={driver.id} value={driver.id}>
                {driver.firstName} {driver.lastName} ({driver.email})
              </option>
            ))}
          </select>
        </div>
      )}

      {assignmentType === 'team' && (
        <div className={styles.teamSelector}>
          <label htmlFor="teamSelect" className={styles.label}>
            Select Team:
          </label>
          <select
            id="teamSelect"
            value={currentTeamId || ''}
            onChange={(e) => onTeamChange(e.target.value || null)}
            className={styles.select}
            disabled={teamsLoading}
          >
            <option value="">Select a team...</option>
            {teams.map((team) => (
              <option key={team.id} value={team.id}>
                {team.name}
              </option>
            ))}
          </select>
        </div>
      )}
    </div>
  );
}
```
