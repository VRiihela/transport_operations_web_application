import { Team, TeamMember, TeamWithMembers } from '../models/Team';
import { DatabaseConnection } from '../database/connection';

export class TeamService {
  constructor(private db: DatabaseConnection) {}

  async getTeamsByDate(date: string): Promise<TeamWithMembers[]> {
    const query = `
      SELECT 
        t.id, t.name, t.date, t.created_by, t.created_at, t.updated_at,
        d.id as driver_id, d.name as driver_name, d.email as driver_email
      FROM teams t
      LEFT JOIN team_members tm ON t.id = tm.team_id
      LEFT JOIN drivers d ON tm.driver_id = d.id
      WHERE t.date = $1
      ORDER BY t.id, d.name
    `;

    const result = await this.db.query(query, [date]);
    return this.groupTeamResults(result.rows);
  }

  async createTeam(name: string, date: string, driverIds: number[], createdBy: number): Promise<TeamWithMembers> {
    const client = await this.db.getClient();
    
    try {
      await client.query('BEGIN');

      // Validate team name
      if (!name || name.trim().length === 0 || name.length > 100) {
        throw new Error('Team name must be 1-100 characters');
      }

      // Validate date
      if (!this.isValidDate(date)) {
        throw new Error('Invalid date format');
      }

      // Check if drivers are already assigned
      const conflictQuery = `
        SELECT DISTINCT tm.driver_id, t.name as team_name
        FROM team_members tm
        JOIN teams t ON tm.team_id = t.id
        WHERE tm.driver_id = ANY($1) AND t.date = $2
      `;
      const conflicts = await client.query(conflictQuery, [driverIds, date]);
      
      if (conflicts.rows.length > 0) {
        const conflictNames = conflicts.rows.map(r => r.team_name).join(', ');
        throw new Error(`Some drivers already assigned to teams: ${conflictNames}`);
      }

      // Create team
      const teamQuery = `
        INSERT INTO teams (name, date, created_by)
        VALUES ($1, $2, $3)
        RETURNING *
      `;
      const teamResult = await client.query(teamQuery, [name.trim(), date, createdBy]);
      const team = teamResult.rows[0];

      // Add team members
      if (driverIds.length > 0) {
        const memberValues = driverIds.map((driverId, index) => 
          `($1, $${index + 2})`
        ).join(', ');
        
        const memberQuery = `
          INSERT INTO team_members (team_id, driver_id)
          VALUES ${memberValues}
        `;
        await client.query(memberQuery, [team.id, ...driverIds]);
      }

      await client.query('COMMIT');

      // Fetch complete team data
      const completeTeam = await this.getTeamById(team.id);
      if (!completeTeam) {
        throw new Error('Failed to retrieve created team');
      }

      return completeTeam;
    } catch (error) {
      await client.query('ROLLBACK');
      throw error;
    } finally {
      client.release();
    }
  }

  async deleteTeam(teamId: number, userId: number): Promise<void> {
    const query = `
      DELETE FROM teams 
      WHERE id = $1 AND created_by = $2
      RETURNING id
    `;
    const result = await this.db.query(query, [teamId, userId]);
    
    if (result.rows.length === 0) {
      throw new Error('Team not found or access denied');
    }
  }

  async getAvailableDrivers(date: string): Promise<Array<{ id: number; name: string; email: string }>> {
    const query = `
      SELECT d.id, d.name, d.email
      FROM drivers d
      WHERE d.id NOT IN (
        SELECT tm.driver_id
        FROM team_members tm
        JOIN teams t ON tm.team_id = t.id
        WHERE t.date = $1
      )
      AND d.active = true
      ORDER BY d.name
    `;
    
    const result = await this.db.query(query, [date]);
    return result.rows;
  }

  private async getTeamById(teamId: number): Promise<TeamWithMembers | null> {
    const query = `
      SELECT 
        t.id, t.name, t.date, t.created_by, t.created_at, t.updated_at,
        d.id as driver_id, d.name as driver_name, d.email as driver_email
      FROM teams t
      LEFT JOIN team_members tm ON t.id = tm.team_id
      LEFT JOIN drivers d ON tm.driver_id = d.id
      WHERE t.id = $1
    `;

    const result = await this.db.query(query, [teamId]);
    const grouped = this.groupTeamResults(result.rows);
    return grouped[0] || null;
  }

  private groupTeamResults(rows: any[]): TeamWithMembers[] {
    const teamsMap = new Map<number, TeamWithMembers>();

    rows.forEach(row => {
      if (!teamsMap.has(row.id)) {
        teamsMap.set(row.id, {
          id: row.id,
          name: row.name,
          date: row.date,
          created_by: row.created_by,
          created_at: row.created_at,
          updated_at: row.updated_at,
          members: []
        });
      }

      if (row.driver_id) {
        teamsMap.get(row.id)!.members.push({
          driver_id: row.driver_id,
          driver_name: row.driver_name,
          driver_email: row.driver_email
        });
      }
    });

    return Array.from(teamsMap.values());
  }

  private isValidDate(date: string): boolean {
    const dateRegex = /^\d{4}-\d{2}-\d{2}$/;
    if (!dateRegex.test(date)) return false;
    
    const parsed = new Date(date);
    return parsed instanceof Date && !isNaN(parsed.getTime());
  }
}

import { Request, Response } from 'express';
import { TeamService } from '../services/TeamService';
import { DatabaseConnection } from '../database/connection';

export class TeamsController {
  private teamService: TeamService;

  constructor(db: DatabaseConnection) {
    this.teamService = new TeamService(db);
  }

  async getTeams(req: Request, res: Response): Promise<void> {
    try {
      const { date } = req.query;

      if (!date || typeof date !== 'string') {
        res.status(400).json({ error: 'Date parameter is required' });
        return;
      }

      const teams = await this.teamService.getTeamsByDate(date);
      res.json({ teams });
    } catch (error) {
      console.error('Error fetching teams:', error);
      res.status(500).json({ error: 'Internal server error' });
    }
  }

  async createTeam(req: Request, res: Response): Promise<void> {
    try {
      const { name, date, driverIds } = req.body;
      const userId = req.user?.id;

      if (!userId) {
        res.status(401).json({ error: 'Unauthorized' });
        return;
      }

      if (!name || !date) {
        res.status(400).json({ error: 'Name and date are required' });
        return;
      }

      if (!Array.isArray(driverIds)) {
        res.status(400).json({ error: 'Driver IDs must be an array' });
        return;
      }

      const team = await this.teamService.createTeam(name, date, driverIds, userId);
      res.status(201).json({ team });
    } catch (error) {
      console.error('Error creating team:', error);
      
      if (error instanceof Error) {
        if (error.message.includes('already assigned') || 
            error.message.includes('Invalid date') ||
            error.message.includes('Team name must be')) {
          res.status(400).json({ error: error.message });
          return;
        }
      }

      res.status(500).json({ error: 'Internal server error' });
    }
  }

  async deleteTeam(req: Request, res: Response): Promise<void> {
    try {
      const { id } = req.params;
      const userId = req.user?.id;

      if (!userId) {
        res.status(401).json({ error: 'Unauthorized' });
        return;
      }

      const teamId = parseInt(id);
      if (isNaN(teamId)) {
        res.status(400).json({ error: 'Invalid team ID' });
        return;
      }

      await this.teamService.deleteTeam(teamId, userId);
      res.status(204).send();
    } catch (error) {
      console.error('Error deleting team:', error);
      
      if (error instanceof Error && error.message.includes('not found')) {
        res.status(404).json({ error: 'Team not found or access denied' });
        return;
      }

      res.status(500).json({ error: 'Internal server error' });
    }
  }

  async getAvailableDrivers(req: Request, res: Response): Promise<void> {
    try {
      const { date } = req.query;

      if (!date || typeof date !== 'string') {
        res.status(400).json({ error: 'Date parameter is required' });
        return;
      }

      const drivers = await this.teamService.getAvailableDrivers(date);
      res.json({ drivers });
    } catch (error) {
      console.error('Error fetching available drivers:', error);
      res.status(500).json({ error: 'Internal server error' });
    }
  }
}

const API_BASE = '/api/teams';

export const teamsApi = {
  async getTeams(date: string): Promise<{ teams: Team[] }> {
    const response = await fetch(`${API_BASE}?date=${encodeURIComponent(date)}`, {
      credentials: 'include'
    });

    if (!response.ok) {
      const error = await response.json();
      throw new Error(error.error || 'Failed to fetch teams');
    }

    return response.json();
  },

  async createTeam(teamData: CreateTeamRequest): Promise<{ team: Team }> {
    const response = await fetch(API_BASE, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json'
      },
      credentials: 'include',
      body: JSON.stringify(teamData)
    });

    if (!response.ok) {
      const error = await response.json();
      throw new Error(error.error || 'Failed to create team');
    }

    return response.json();
  },

  async deleteTeam(teamId: number): Promise<void> {
    const response = await fetch(`${API_BASE}/${teamId}`, {
      method: 'DELETE',
      credentials: 'include'
    });

    if (!response.ok) {
      const error = await response.json();
      throw new Error(error.error || 'Failed to delete team');
    }
  },

  async getAvailableDrivers(date: string): Promise<{ drivers: Driver[] }> {
    const response = await fetch(`${API_BASE}/available-drivers?date=${encodeURIComponent(date)}`, {
      credentials: 'include'
    });

    if (!response.ok) {
      const error = await response.json();
      throw new Error(error.error || 'Failed to fetch available drivers');
    }

    return response.json();
  }
};

import { useState, useEffect, useCallback } from 'react';
import { Team, Driver } from '../types/Team';
import { teamsApi } from '../api/teams';

export const useTeams = (selectedDate: string) => {
  const [teams, setTeams] = useState<Team[]>([]);
  const [availableDrivers, setAvailableDrivers] = useState<Driver[]>([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const fetchTeams = useCallback(async () => {
    if (!selectedDate) return;

    setLoading(true);
    setError(null);

    try {
      const [teamsResult, driversResult] = await Promise.all([
        teamsApi.getTeams(selectedDate),
        teamsApi.getAvailableDrivers(selectedDate)
      ]);

      setTeams(teamsResult.teams);
      setAvailableDrivers(driversResult.drivers);
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Failed to fetch teams');
    } finally {
      setLoading(false);
    }
  }, [selectedDate]);

  useEffect(() => {
    fetchTeams();
  }, [fetchTeams]);

  const createTeam = async (name: string, driverIds: number[]) => {
    try {
      setError(null);
      const result = await teamsApi.createTeam({
        name,
        date: selectedDate,
        driverIds
      });

      setTeams(prev => [...prev, result.team]);
      await fetchTeams(); // Refresh available drivers
      return result.team;
    } catch (err) {
      const errorMessage = err instanceof Error ? err.message : 'Failed to create team';
      setError(errorMessage);
      throw new Error(errorMessage);
    }
  };

  const deleteTeam = async (teamId: number) => {
    try {
      setError(null);
      await teamsApi.deleteTeam(teamId);
      setTeams(prev => prev.filter(team => team.id !== teamId));
      await fetchTeams(); // Refresh available drivers
    } catch (err) {
      const errorMessage = err instanceof Error ? err.message : 'Failed to delete team';
      setError(errorMessage);
      throw new Error(errorMessage);
    }
  };

  return {
    teams,
    availableDrivers,
    loading,
    error,
    createTeam,
    deleteTeam,
    refetch: fetchTeams
  };
};
